385  int retval;

394 inline int mutabor_get_note_distance(int note, tone_system * system) {
395  int delta, retval;
396  if (!system) return INT_MIN;
397  delta = (note - system->anker);
398  retval = delta / system->breite;
399  if (retval * system->breite > delta)
400  retval -= 1;
401  mutASSERT(system -> anker + retval * system -> breite <= note);
402  mutASSERT(system -> anker + (retval + 1) * system -> breite > note);
403  return retval;
