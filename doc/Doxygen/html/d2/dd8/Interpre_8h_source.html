291 inline mutabor_tone mutabor_add_interval_to_tone(const mutabor_tone tone,
292  int count,
293  const mutabor_interval interval) {
294  mutabor_tone retval;
295  retval.active = tone.active;
296  retval.value = tone.value + ((mutint64) count) * interval.value;
297  switch (interval.active) {
298  case mutabor_empty_interval:
299  if (tone.active == mutabor_invalid_interval)
300  break;
301  FALLTHROUGH;
302  case mutabor_invalid_interval:
303  retval.active = interval.active;
304  break;
305  case mutabor_active_interval:
306  /* do nothing */
307  break;
308  }
309  return retval;
310 }

363 inline mutabor_tone mutabor_tone_copy_distance(const mutabor_tone offset,
364  const mutabor_tone t1,
365  const mutabor_tone t2) {
366  mutabor_tone retval;
367  if (offset.active == mutabor_invalid_tone ||
368  t1.active == mutabor_invalid_tone ||
369  t2.active == mutabor_invalid_tone)
370  retval.active = mutabor_invalid_tone;
371  else if (offset.active == mutabor_empty_tone ||
372  t1.active == mutabor_empty_tone ||
373  t2.active == mutabor_empty_tone)
374  retval.active = mutabor_empty_tone;
375  else if (offset.active == mutabor_active_tone &&
376  t1.active == mutabor_active_tone &&
377  t2.active == mutabor_active_tone)
378  retval.active = mutabor_active_tone;
379  else
380  retval.active = mutabor_invalid_tone;
381  retval.value = offset.value + (t1.value - t2.value);
382  return retval;
383 }

385 inline int mutabor_get_note_index(int note, tone_system * system) {
386  int retval;
387  if (!system) return INT_MIN;
388  retval = (note - system->anker) % system->breite;
389  if (retval < 0) retval += system->breite;
390  mutASSERT(0 <= retval);
391  mutASSERT(retval < system->breite);
392  return retval;
393 }

395 inline int mutabor_get_note_distance(int note, tone_system * system) {
396  int delta, retval;
397  if (!system) return INT_MIN;
398  delta = (note - system->anker);
399  retval = delta / system->breite;
400  if (retval * system->breite > delta)
401  retval -= 1;
402  mutASSERT(system -> anker + retval * system -> breite <= note);
403  mutASSERT(system -> anker + (retval + 1) * system -> breite > note);
404  return retval;
405 }
