75  void BoxData::set_routefile_id(int id) {
76  BoxClass::set_routefile_id(id);
77  if (box)
78  id = box->id;
79 
80  switch (id) {
81  case mutabor::NewBox:
82  case mutabor::NoBox:
83  case mutabor::GmnBox:
84  background_colour = wxNullColour;
85  text_colour = *wxBLACK;
86  break;
87  default:
88  int r = ((id & 0x01) << 7 ) | ((id & 0x08) << 3) | ((id & 0x40) >> 1) |
89  ((id & 0x200) >> 5) | ((id & 0x1000) >> 9) | ((id & 0x8000) >> 13) |
90  ((id & 0x40000) >> 17) | ((id & 0x200000) >> 21) ;
91  r = r?r-1:0;
92  int g = ((id & 0x02) << 6 ) | ((id & 0x10) << 2) | ((id & 0x80) >> 2) |
93  ((id & 0x400) >> 6) | ((id & 0x2000) >> 10) | ((id & 0x10000) >> 14) |
94  ((id & 0x80000) >> 18) | ((id & 0x400000) >> 22);
95  g = g?g-1:0;
96  int b = ((id & 0x04) << 5 ) | ((id & 0x20) << 1) | ((id & 0x100) >> 3) |
97  ((id & 0x800) >> 7) | ((id & 0x4000) >> 11) | ((id & 0x20000) >> 15) |
98  ((id & 0x100000) >> 19) | ((id & 0x800000) >> 23);
99  b = b?b-1:0;
100  DEBUGLOG2(other,_T("Box %d color %x,%x,%x"),id-1,r,g,b);
101  if (r+b+g < 0x180)
102  text_colour = *wxWHITE;
103  else
104  text_colour = *wxBLACK;
105  background_colour = wxColour(r,g,b);
106  }
107  MutChildApi * windows[] = { winattr.key_window,
108  winattr.tonesystem_window,
109  winattr.actions_window,
110  winattr.logic_window };
