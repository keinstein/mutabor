227 void berechne_intervalle_absolut (mutabor_box_type * box, struct intervall * list_of_intervalle)
228 
229 {
230  size_t i,j;
231 
232  box->file->interval_count = intervall_list_laenge (list_of_intervalle);
233 
234  box->file->check_intervals = (struct intervall* *) xalloca (box, sizeof(struct intervall *) * box->file->interval_count);
235  box->file->visited_intervals = (char*) xalloca (box, sizeof(char) * box->file->interval_count);
236  box->file->interval_matrix = (char*) xalloca (box, sizeof(char) * box->file->interval_count * box->file->interval_count);
237 
238 
239  /* Feld mit intervallen initialisieren (zum schnelleren Zugriff) */
240 
241  belege_intervalle (box->file->check_intervals, list_of_intervalle);
242 
243  /* Adjazenzmatrix initialisieren (Kein Intervall hängt vom anderen ab) */
244 
245  for (i=0; i<box->file->interval_count; i++)
246  {
247  for (j=0; j<box->file->interval_count; j++) {
248  adjacent (box->file, i,j) = 0;
249  }
250  }
251 
252  /* Adjazenzmatrix initialisieren (Abhängigkeiten eintragen) */
253 
254  for (i=0; i<box->file->interval_count; i++)
255  {
256  if (box->file->check_intervals[i]->intervall_typ == intervall_absolut) /* alles ok */ ;
257  else if (box->file->check_intervals[i]->intervall_typ == intervall_komplex) {
258 
259  struct komplex_intervall * lauf;
260 
261  for (lauf = box->file->check_intervals[i]->u.intervall_komplex.komplex_liste;
262  lauf;
263  lauf = lauf -> next) {
264  adjacent (box->file, i, intervall_nummer (box, lauf -> name)) = 1;
265  }
266  } else {
267  mutabor_error_message(box,
268  internal_error,
269  _("Undefined interval type %d for interval %s."),
270  box->file->check_intervals[i]->intervall_typ,
271  (box->file->check_intervals[i]->name));
272  }
273  }
274 
275 #ifdef DEBUG_ANZEIGE_3
276  /* Adjazenzmatrix anzeigen */
277 
278  printf ("Matrix:\n");
279 
280  for (i=0; i<box->file->interval_count; i++)
281  {
282  printf ("%s -> ", box->file->check_intervals[i]->name);
283 
284  for (j=0; j<box->file->interval_count; j++) {
285  if (adjacent (box->file, i,j))
286  printf ("%s ", box->file->check_intervals[j]->name);

300 
301  for (i=0; i<box->file->interval_count; i++)

306  }
307
