506  void InputMidiPort::Proceed(const std::vector<unsigned char > &midiCode, int data, int channel_offset) {
508  uint8_t MidiChannel = (midiCode[0] & 0x0F) + channel_offset;
509  uint8_t MidiStatus = midiCode[0];
510  DEBUGLOG (midifile, "Status: %x" , MidiStatus);
511 
512  switch ( MidiStatus ) {
513  case midi::SYSTEM_UNDEFINED1:
514  case midi::SYSTEM_UNDEFINED2:
515  case midi::SYSEX_END:
516  case midi::REALTIME_UNDEFINED:
517  case midi::ACTIVE_SENSE:
518  /* unimplemented or intentionally ignored */
519  return;
520 
521  case midi::STOP_PLAY:{
522  InputDeviceClass::PauseAll();
523  event e = create_event(midiCode, MidiChannel);
524  OutputDeviceClass::all_handle_event(e);
525  }
526  return;
527  case midi::START_PLAY:
528  InputDeviceClass::StopAll();
529  FALLTHROUGH;
530  case midi::CONTINUE_PLAY: {
531  event e = create_event(midiCode, MidiChannel);
532  OutputDeviceClass::all_handle_event(e);
533  InputDeviceClass::RealtimePlay();
534  }
535  return;
536  case midi::CLOCK:
537  case midi::TICK:
538  case midi::SONG_SELECT:
539  case midi::SONG_POSITION:
540  case midi::QUARTER_FRAME: {
541  event e = create_event(midiCode,
542  MidiChannel);
543  OutputDeviceClass::all_handle_event(e);
544  }
545  return;
546  // the following are handled by midicmn.cpp
547  case midi::TUNE_REQUEST:
548  case midi::META:
549  case midi::SYSEX_START:
550  default:
551  break;
552  }
553 
554  parentType::Proceed(midiCode, data, channel_offset);
555  }
