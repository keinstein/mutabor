450  {
451  mutASSERT(midiCode);
452  switch ( R->GetType() ) {
453  case RTchannel:

457  case RTstaff:

475  void InputMidiPort::Proceed(const std::vector<unsigned char > * midiCode, int data, int channel_offset) {
477  uint8_t MidiChannel = (midiCode->at(0) & 0x0F) + channel_offset;
478  uint8_t MidiStatus = midiCode->at(0);
479  DEBUGLOG (midifile, "Status: %x" , MidiStatus);
480 
481  switch ( MidiStatus ) {
482  case midi::SYSTEM_UNDEFINED1:
483  case midi::SYSTEM_UNDEFINED2:
484  case midi::SYSEX_END:
485  case midi::REALTIME_UNDEFINED:
486  case midi::ACTIVE_SENSE:
487  /* unimplemented or intentionally ignored */
488  return;
489 
490  case midi::STOP_PLAY:{
491  InputDeviceClass::PauseAll();
492  event e = create_event(midiCode, MidiChannel);
493  OutputDeviceClass::all_handle_event(e);
494  }
495  return;
496  case midi::START_PLAY:
497  InputDeviceClass::StopAll();
498  case midi::CONTINUE_PLAY: {
499  event e = create_event(midiCode, MidiChannel);
500  OutputDeviceClass::all_handle_event(e);
501  InputDeviceClass::RealtimePlay();
502  }
503  return;
504  case midi::CLOCK:
505  case midi::TICK:
506  case midi::SONG_SELECT:
507  case midi::SONG_POSITION:
508  case midi::QUARTER_FRAME: {
509  event e = create_event(midiCode,
510  MidiChannel);
511  OutputDeviceClass::all_handle_event(e);
512  }
513  return;
514  // the following are handled by midicmn.cpp
515  case midi::TUNE_REQUEST:
516  case midi::META:
517  case midi::SYSEX_START:
518  default:
519  break;
520  }
521 
522  parentType::Proceed(midiCode, data, channel_offset);
523  }
