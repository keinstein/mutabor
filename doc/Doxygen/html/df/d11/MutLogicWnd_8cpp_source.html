411  void MutLogicWnd::UpDate()
412  {
413  if (Ok) return;
414  /* We might use a mutex here until rendering is done
415  to avoid rerendering, but this would at least slow
416  down music handling in BoxChangedAction. Currently
417  we waste time in the main thread when it is idle
418  anyway.
419  */
420  Ok = true;
421  // Analyse zuerst
422  BoxData * guibox = ToGUIBase(box);
423  mutASSERT(guibox);
424 
425  wxSizer * sizer = GetSizer();
426  wxSizerFlags flags;
427  flags.Expand().Centre();
428 
429 
430  //* \todo we must reuse old entries as rendering can be expensive
431  DestroyChildren();
432  MutTag * toFocus=NULL;
433  bool isOpen;
434 
435  mutabor::BoxClass::logic_list logics = box->GetLogics();
436  for (mutabor::BoxClass::logic_list::iterator i = logics.begin();
437  i != logics.end();
438  i++) {
439  if ( (isOpen = i->flags == mutabor::BoxClass::logic_entry::CurrentLogic)) {
440 #if 0
441  // guibox->SetLogic(sText);
442  if ( !sEinst.IsEmpty() )
443  guibox->SetTonesystem(sEinst);
444  else if ( guibox->GetTonesystem().empty() )
445  guibox->SetTonesystem(_("(INITIAL)"));
446  else if ( guibox->GetTonesystem()[0] != '[' )
447  guibox->SetTonesystem(
448  _T("[")
449  + guibox->GetTonesystem()
450  + _T("]")
451  );
452 #endif
453  } else
454  // guibox->SetTonesystem(sText);
455  ;
456  bool isLogic = isOpen||i->flags == mutabor::BoxClass::logic_entry::Logic;
457  MutTag * tag = new MutTag(this,
458  wxDefaultPosition,
459  isLogic,
460  isOpen || i->active,
461  i -> key);
462  std::string taglabel = isLogic ? i-> name + (!i->startTuning.empty() ?
463  "\n(" + i -> startTuning + ")":""):
464  i->startTuning;
465 #if wxCHECK_VERSION(2,9,0)

471  tag -> SetForegroundColour(GetForegroundColour());
472  TagMinSize.IncTo(tag->GetBestSize());
473  if (sizer)
474  sizer->Add(tag,flags);
475  if (isOpen || i->active) toFocus = tag;
476  }

478  Layout();
479  FixSizer();
