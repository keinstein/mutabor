110 
111  for (; current != t.tones.end()
112  && current->flag != mutabor::box_support::tone_entry::sounding;

114  ++reference;
115  keys.push_back(key());
116  }
117

120  reference_frequency = std::exp((current->pitch - 69.0)/12.0*log(2.0l)) * 440;
121

123  for(; current != t.tones.end(); ++current) {
124  if (current->flag != mutabor::box_support::tone_entry::sounding) {
125  keys.push_back(key());
126  } else {
127  keys.push_back(interval++);
128  }
129  }
130  count = keys.size();
131  repetition_interval = interval;
