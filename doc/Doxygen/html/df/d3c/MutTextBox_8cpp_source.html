294 void MutTextBox::GetToneSystem(bool asTS)
295 {
296  wxString keys;
297 
298  mutabor::BoxClass::tone_system tonesys = box->GetToneSystem();
299  Freeze();
300 
301  Clear();
302 
303 
304  keys.Printf(_("Anchor = %d"), tonesys.anchor);
305  Append(keys);
306  keys.Printf(_("Width = %u"), (unsigned)tonesys.tones.size());
307  Append(keys);
308  keys.Printf(_("Period = %.1f HT"), tonesys.period);
309  Append(keys);
310 
311  if (tonesys.tones.size() < 1) {
312  Append(_("No tones."));
313  Thaw();
314  return;
315  }
316  size_t i;
317  double reference = 0.0l;
318  if (!asTS) {
319  for (i = 0 ; i < tonesys.tones.size(); i++) {
320  if (tonesys.tones[i].flag ==
321  mutabor::BoxClass::tone_entry::sounding) {
322  reference = tonesys.tones[i].pitch;
323  break;
324  }
325  }
326  keys.Printf(_("Reference = %d HT"),i);
327  Append(keys);
328  }
329 
330  for (i = 0 ; i < tonesys.tones.size(); i++) {
331  double pitch, frequency;
332  switch(tonesys.tones[i].flag) {
333  case mutabor::BoxClass::tone_entry::sounding:
334  pitch = tonesys.tones[i].pitch;
335  frequency =
336  mutabor_convert_pitch_to_frequency(pitch);
337  if (!asTS) pitch -= reference;
338  keys.Printf(_("%2u : %8.1lf Hz (%6.2lf HT)"),
339  (unsigned)i,
340  frequency,
341  pitch );
342  break;
343  case mutabor::BoxClass::tone_entry::silent:
344  keys.Printf(_("%2d : empty"),i);
345  break;
346  case mutabor::BoxClass::tone_entry::invalid:
347  keys.Printf(_("%2d : invalid tone"),i);
348  break;
349  default:
350  keys.Printf(_("%2d : (unknown type %d)"),
351  i,tonesys.tones[i].flag);
352  }
353  Append(keys);
354  }
355  Thaw();
