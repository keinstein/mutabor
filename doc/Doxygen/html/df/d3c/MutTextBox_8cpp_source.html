301 void MutTextBox::GetToneSystem(bool asTS)
302 {
303  wxString keys;
304 
305  mutabor::BoxClass::tone_system tonesys = box->GetToneSystem();
306  Freeze();
307 
308  Clear();
309 
310 
311  keys.Printf(_("Anchor = %d"), tonesys.anchor);
312  Append(keys);
313  keys.Printf(_("Width = %u"), (unsigned)tonesys.tones.size());
314  Append(keys);
315  keys.Printf(_("Period = %.1f HT"), tonesys.period);
316  Append(keys);
317 
318  if (tonesys.tones.size() < 1) {
319  Append(_("No tones."));
320  Thaw();
321  return;
322  }
323  size_t i;
324  double reference = 0.0l;
325  if (!asTS) {
326  for (i = 0 ; i < tonesys.tones.size(); i++) {
327  if (tonesys.tones[i].flag ==
328  mutabor::BoxClass::tone_entry::sounding) {
329  reference = tonesys.tones[i].pitch;
330  break;
331  }
332  }
333  keys.Printf(_("Reference = %d HT"),i);
334  Append(keys);
335  }
336 
337  for (i = 0 ; i < tonesys.tones.size(); i++) {
338  double pitch, frequency;
339  switch(tonesys.tones[i].flag) {
340  case mutabor::BoxClass::tone_entry::sounding:
341  pitch = tonesys.tones[i].pitch;
342  frequency =
343  mutabor_convert_pitch_to_frequency(pitch);
344  if (!asTS) pitch -= reference;
345  keys.Printf(_("%2u : %8.1lf Hz (%6.2lf HT)"),
346  (unsigned)i,
347  frequency,
348  pitch );
349  break;
350  case mutabor::BoxClass::tone_entry::silent:
351  keys.Printf(_("%2d : empty"),i);
352  break;
353  case mutabor::BoxClass::tone_entry::invalid:
354  keys.Printf(_("%2d : invalid tone"),i);
355  break;
356  default:
357  keys.Printf(_("%2d : (unknown type %d)"),
358  i,tonesys.tones[i].flag);
359  }
360  Append(keys);
361  }
