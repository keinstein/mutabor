192 inline static void call_actions (mutabor_box_type * box,
193  struct do_aktion * aktion_list,
194  struct interpreter_parameter_list * parameters) {
195  interpreter_parameter_list * current_parameters;
196  interpreter_parameter_list * old_parameters = box->current_parameters;
197 
198  if (!box || box->flags.break_logic) return;
199 
200  // check whether we must allocate memory
201  if ((box->current_parameters == NULL && box->parameters == NULL)
202  || (box->current_parameters != NULL
203  && box->current_parameters->next == NULL)) {
204  current_parameters = (struct interpreter_parameter_list *)
205  ymalloc(box,sizeof(struct interpreter_parameter_list));
206  current_parameters->next = NULL;
207  current_parameters->data = (int *)ycalloc(box,
208  box->file->parameter_count,
209  sizeof(int));
210  if (box->parameters == NULL) {
211  mutASSERT(box->current_parameters == NULL);
212  box->current_parameters =
213  box->parameters = current_parameters;
214  } else {
215  box->current_parameters->next = current_parameters;
216  }
217 
218  }
219  // we have a free parameter list
220  else if ( box->current_parameters == NULL) {
221  current_parameters = box->current_parameters = box->parameters;
222  } else {
223  current_parameters = box->current_parameters -> next;
224  box->current_parameters = current_parameters;
225  }
226  for (struct do_aktion * aktion = aktion_list ;
227  aktion; aktion = aktion -> next) {
228  interpreter_argument_list * arguments =
229  aktion -> secondary_arguments ?
230  aktion -> secondary_arguments :
231  aktion -> arguments;
232  size_t size;
233  if (arguments == NULL) {
234  execute_aktion (box, aktion, NULL);
235  continue;
236  }
237  size = arguments->size;
238  current_parameters -> size = size;
239  mutASSERT(size <= box->file->parameter_count);
240  for (size_t i = 0 ; i < size; i++) {
241  switch (arguments->types[i]) {
242  case mutabor_argument_integer:
243  current_parameters->data[i] = arguments->data[i].constant;
244  break;
245  case mutabor_argument_parameter:
246  current_parameters->data[i]
247  = parameters->data[arguments->data[i].index];
248  break;
249  case mutabor_argument_distance:
250  current_parameters->data[i]
251  = box->distance;
252  break;
253  case mutabor_argument_anchor:
254  current_parameters->data[i]
255  = box->tonesystem->anker;
256  break;
257  case mutabor_argument_invalid:
258  current_parameters->data[i] = -1;
259  mutASSERT(arguments->types[i] != mutabor_argument_invalid);
260  break;
261  }
262 
263  }
264  execute_aktion(box,
265  aktion,
266  current_parameters);
267  }
268  if (old_parameters != NULL)
269  box->current_parameters = old_parameters;
270 }

721  void change_anker(mutabor_box_type * box, int neu)
722  {
723  tone_system * tonsys = box->tonesystem;
724  tone_system * free_tonesystem =
725  box->tonesystem == box->tonesystem_memory ?
726  &(box->tonesystem_memory[1]) :
727  &(box->tonesystem_memory[0]);
728  int i;
729  TRACE;
730  /* update the distance parameter for logic calls */
731  box->distance -= neu-tonsys->anker;
732 
733  while ( box->distance < 0 )
734  box->distance += tonsys->breite;
735 
736  box->distance %= tonsys->breite;
737 
738  /* update the tone system */
739  while (neu< MUTABOR_KEYRANGE_MIN_KEY)
740  neu += tonsys->breite;
741 
742  while (neu> MUTABOR_KEYRANGE_MAX_KEY)
743  neu -= tonsys->breite;
744 
745  free_tonesystem->ton[0]=mutabor_get_note_frequency(neu,tonsys);
746 
747  for (i=1; i<tonsys->breite; i++)
748  free_tonesystem->ton[i] =
749  mutabor_tone_copy_distance(free_tonesystem->ton[0],
750  tonsys->ton[i],
751  tonsys->ton[0]);
752 
753  free_tonesystem->anker=neu;
754  free_tonesystem->breite=tonsys->breite;
755  free_tonesystem->periode=tonsys->periode;
756  box->tonesystem = free_tonesystem;
