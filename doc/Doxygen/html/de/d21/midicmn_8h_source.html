128  iterator reserve_channel(iterator actual) {
129  if (actual != this->end()) {
130  if (first_free <= actual) {
131  move_to_front(actual,first_free);
132  actual = first_free;

732  msb or lsb only. Otherwise we do not have access to the internal state
733  of the devie and leave it alone with the setting the user provided */
734  // check for valid msb/lsb combinations

837  long fix_bend(long bend) {
838  long retval;
839  if (bend < 0) {
840  retval = bend / bending_range;
841  } else {
842  mutint64 b = bend * (mutint64)0x1fff;
843  retval = b / (0x2000 * bending_range);
844 
845  }
846 
847  /* A naive approach assumes that pitch bend ranges
848  from -0x2000 to 0x2000. Actually the range lasts
849  from -0x2000 to 0x1fff.
850 
851  The difference is one step or
852  (pitch_bend_range / 0x1fff). For 12HT pitch bend range
853  this means about 0.3ct.
854 
855  Note: The auditory system has about 4000 receptors,
856  which proviedes an average distance of 3ct. But they
857  are not equally distributed and the amplification
858  is not linear, which makes difference tones to be real,
859  detectable frequencies.
860 
861  In absense of the standard (it's at work) I
862  implemented the complicate correct formula above.
863  */
864 
865  // must solve the equation 0x1 00 00 00 / divisor + 0x 3f ff = 0x 7f ff
866  return retval / 0x800; //we can't use >> because of negative values
867  }
