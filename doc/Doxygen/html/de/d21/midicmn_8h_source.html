129  iterator reserve_channel(iterator actual) {

733  DEBUGLOG (midiio, "ch: %d, pr: %d, lsb: %d, msb: %d, send mode: %d" , channel, sound, lsb, msb, bank_mode);
734  /* if only one bank byte has been set we correct it in case we send
735  msb or lsb only. Otherwise we do not have access to the internal state
736  of the devie and leave it alone with the setting the user provided */
737  // check for valid msb/lsb combinations
738  if (!(lsb & 0x80)) {
739  if (!(msb & 0x80)) {
740  switch (bank_mode) {
741  case lsb_first:
742  controller(channel,midi::BANK_FINE,lsb);
743  FALLTHROUGH;
744  case msb_only:
745  controller(channel,midi::BANK_COARSE,msb);
746  break;
747  case msb_first:
748  controller(channel,midi::BANK_COARSE,msb);
749  FALLTHROUGH;
750  case lsb_only:
751  controller(channel,midi::BANK_FINE,lsb);
752  break;
753  default:
754  UNREACHABLEC;
755  }
756  } else {
757  controller(channel,(bank_mode == msb_only?
758  midi::BANK_COARSE:
759  midi::BANK_FINE),lsb);
760  }
761  } else {
762  if (!(msb & 0x80)) {
763  controller(channel,(bank_mode == lsb_only?
764  midi::BANK_FINE:
765  midi::BANK_COARSE),lsb);
766  }
767  }

843  long fix_bend(long bend) {
844  long retval = bend / bending_range;
845 
846  // Midi Pitch bend has 13 bit precision
847  // we have 24 bit precision
848  // 0x1000000 / 0x2000
849  constexpr long divisor = BoxClass::tone::get_semitone() / 0x2000;
850  constexpr long divisor2 = divisor >> 1;
851  // in order to get correct rounding we add
852  // half of the
853 
854  long tmp = retval % divisor;
855  retval = retval / divisor;
856  if (tmp > divisor2)
857  retval += 1;
858  else if (tmp < -divisor2)
859  retval -= 1;
860 
861  /* A naive approach assumes that pitch bend ranges
862  from -0x2000 to 0x2000. Actually the range lasts
863  from -0x2000 to 0x1fff.
864 
865  The difference is one step or
866  (pitch_bend_range / 0x1fff). For 12HT pitch bend range
867  this means about 0.3ct.
868 
869  Note: The auditory system has about 4000 receptors,
870  which proviedes an average distance of 3ct. But they
871  are not equally distributed and the amplification
872  is not linear, which makes difference tones to be real,
873  detectable frequencies.
874 
875  In absense of the standard (it's at work) I
876  implemented the complicate correct formula above.
877  */
878 
879  // must solve the equation 0x1 00 00 00 / divisor + 0x 3f ff = 0x 7f ff
880  return retval;
881  }

883  pitch_bend_type pitch_and_bend(const BoxClass::tone & tuned_note) {
884  pitch_bend_type data;
885  data.bend = tuned_note.get_bend();
886  data.pitch = tuned_note.get_pitch();
887  if (data.bend >= tuned_note.get_quartertone()) {
888  data.pitch++;
889  data.bend -= tuned_note.get_semitone();
890  }

892  // must solve the equation 0x 80 00 00 / divisor + 0x 40 00 = 0x 1 00 00
893  data.bend = fix_bend(data.bend); //we can't use >> because of negative values
