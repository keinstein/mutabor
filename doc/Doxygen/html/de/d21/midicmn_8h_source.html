128  iterator reserve_channel(iterator actual) {
129  if (actual != this->end()) {
130  if (first_free <= actual) {
131  move_to_front(actual,first_free);
132  actual = first_free;
133  ++first_free;
134  if (first_free > last_sustained) {
135  ++last_sustained;

141  move_to_end(actual, first_free);
142  actual = first_free - 1;

721  long fix_bend(long bend) {
722  long retval;
723  if (bend < 0) {
724  retval = bend / bending_range;
725  } else {
726  mutint64 b = bend * (mutint64)0x1fff;
727  retval = b / (0x2000 * bending_range);
728 
729  }
730 
731  /* A naive approach assumes that pitch bend ranges
732  from -0x2000 to 0x2000. Actually the range lasts
733  from -0x2000 to 0x1fff.
734 
735  The difference is one step or
736  (pitch_bend_range / 0x1fff). For 12HT pitch bend range
737  this means about 0.3ct.
738 
739  Note: The auditory system has about 4000 receptors,
740  which proviedes an average distance of 3ct. But they
741  are not equally distributed and the amplification
742  is not linear, which makes difference tones to be real,
743  detectable frequencies.
744 
745  In absense of the standard (it's at work) I
746  implemented the complicate correct formula above.
747  */
748 
749  // must solve the equation 0x1 00 00 00 / divisor + 0x 3f ff = 0x 7f ff
750  return retval / 0x800; //we can't use >> because of negative values
751  }

754  pitch_bend_type data;
755  data.bend = tuned_note & 0xffffff;
756  data.pitch = tuned_note >> 24;
