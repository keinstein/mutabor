159  int set_controller(size_t number, int data) {
160 
161  if (controller.size() <= number)
162  controller.resize(number +1,-1);
163  int retval = controller[number];
164  int param = -1;
165  DEBUGLOG(midiio,("ctrl: %d, %d => %d"),(int)number,(int)retval,(int)data);
166  controller[number] = data;
167 
168  // we allow to set a controller to an udefined state.
169  if (data < 0) return retval;
170 
171  switch (number) {
172  case midi::NON_REGISTERED_PARAMETER_FINE:
173  case midi::NON_REGISTERED_PARAMETER_COARSE:
174  data_is_rpn = false;
175  break;
176  case midi::REGISTERED_PARAMETER_FINE:
177  case midi::REGISTERED_PARAMETER_COARSE:
178  data_is_rpn = true;
179  break;
180  case midi::DATA_ENTRY_COARSE:
181  case midi::DATA_ENTRY_FINE:
182  case midi::DATA_BUTTON_INCREMENT:
183  case midi::DATA_BUTTON_DECREMENT:
184  if (!data_is_rpn
185  && controller[midi::NON_REGISTERED_PARAMETER_FINE] != -1
186  && controller[midi::NON_REGISTERED_PARAMETER_COARSE] != -1
187  && controller[midi::NON_REGISTERED_PARAMETER_FINE] != 0x7f
188  && controller[midi::NON_REGISTERED_PARAMETER_COARSE] != 0x7f) {
189  param = midi::FIRST_NRPN
190  | controller[midi::NON_REGISTERED_PARAMETER_COARSE] << 8
191  | controller[midi::NON_REGISTERED_PARAMETER_FINE];
192  } else if (data_is_rpn
193  && controller[midi::REGISTERED_PARAMETER_FINE] != -1
194  && controller[midi::REGISTERED_PARAMETER_COARSE] != -1
195  && controller[midi::REGISTERED_PARAMETER_FINE] != 0x7f
196  && controller[midi::REGISTERED_PARAMETER_COARSE] != 0x7f) {
197  param = midi::FIRST_RPN
198  | controller[midi::REGISTERED_PARAMETER_COARSE] << 8
199  | controller[midi::REGISTERED_PARAMETER_FINE];
200  }
201  if (param != -1) {
202  if (controller.size() <= (size_t) param)
203  controller.resize(param+1,-1);
204  switch (number) {
205  case midi::DATA_BUTTON_INCREMENT:
206  if (!(controller[param]&0x80)) {
207  // skip the sign bit for lsb
208  int value = controller[param];
209  value++;
210  if (value & 0x80) {
211  if (!(value & 0x8000))
212  value += 0x100;
213  value &= ~((int)0xff);
214  }
215  controller[param] = value;
216  }
217  break;
218  case midi::DATA_BUTTON_DECREMENT:
219  if (!(controller[param]&0x80)) {
220  int value = controller[param];
221  bool hasmsb = !(value & 0x8000);
222  value--;
223  if (value & 0x80) {
224  if (!hasmsb) {
225  value |= 0xFF00;
226  }
227  value &= ~((int)0x80);
228  }
229  }
230  break;
231  case midi::DATA_ENTRY_COARSE: {
232  int value = (controller[param] & 0xFF) | ((data & 0xFF) << 8);
233  if (controller[param] != value)
234  set_controller(param,value);
235  }
236  break;
237  case midi::DATA_ENTRY_FINE: {
238  int value = (controller[param] & 0xff00)| (data & 0xff);
239  if (controller[param] != value)
240  set_controller(param,value);
241  }
242  break;
243  }
244  }
245  break;
246  }
