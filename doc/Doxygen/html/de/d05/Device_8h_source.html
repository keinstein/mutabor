162  controller[number] = data;
163 
164  // we allow to set a controller to an udefined state.
165  if (data < 0) return retval;
166 
167  switch (number) {
168  case midi::NON_REGISTERED_PARAMETER_FINE:
169  case midi::NON_REGISTERED_PARAMETER_COARSE:
170  data_is_rpn = false;
171  break;
172  case midi::REGISTERED_PARAMETER_FINE:
173  case midi::REGISTERED_PARAMETER_COARSE:
174  data_is_rpn = true;
175  break;
176  case midi::DATA_ENTRY_COARSE:
177  case midi::DATA_ENTRY_FINE:
178  case midi::DATA_BUTTON_INCREMENT:
179  case midi::DATA_BUTTON_DECREMENT:
180  if (!data_is_rpn
181  && controller[midi::NON_REGISTERED_PARAMETER_FINE] != -1
182  && controller[midi::NON_REGISTERED_PARAMETER_COARSE] != -1
183  && controller[midi::NON_REGISTERED_PARAMETER_FINE] != 0x7f
184  && controller[midi::NON_REGISTERED_PARAMETER_COARSE] != 0x7f) {
185  param = midi::FIRST_NRPN
186  | controller[midi::NON_REGISTERED_PARAMETER_COARSE] << 8
187  | controller[midi::NON_REGISTERED_PARAMETER_FINE];
188  } else if (data_is_rpn

190  && controller[midi::REGISTERED_PARAMETER_COARSE] != -1
191  && controller[midi::REGISTERED_PARAMETER_FINE] != 0x7f
192  && controller[midi::REGISTERED_PARAMETER_COARSE] != 0x7f) {
193  param = midi::FIRST_RPN
194  | controller[midi::REGISTERED_PARAMETER_COARSE] << 8
195  | controller[midi::REGISTERED_PARAMETER_FINE];
196  }
197  if (param != -1) {
198  if (controller.size() <= (size_t) param)
199  controller.resize(param+1,-1);
200  switch (number) {
201  case midi::DATA_BUTTON_INCREMENT:
202  if (!(controller[param]&0x80)) {
203  // skip the sign bit for lsb
204  int value = controller[param];
205  value++;
206  if (value & 0x80) {
207  if (!(value & 0x8000))
208  value += 0x100;
209  value &= ~((int)0xff);
210  }
211  controller[param] = value;
212  }
213  break;
214  case midi::DATA_BUTTON_DECREMENT:
215  if (!(controller[param]&0x80)) {
216  int value = controller[param];
217  bool hasmsb = !(value & 0x8000);
218  value--;
219  if (value & 0x80) {
220  if (!hasmsb) {
221  value |= 0xFF00;
222  }
223  value &= ~((int)0x80);
224  }
225  }
226  break;
227  case midi::DATA_ENTRY_COARSE: {
228  int value = (controller[param] & 0xFF) | ((data & 0xFF) << 8);
229  if (controller[param] != value)
230  set_controller(param,value);
231  }
232  break;
233  case midi::DATA_ENTRY_FINE: {
234  int value = (controller[param] & 0xff00)| (data & 0xff);
235  if (controller[param] != value)
236  set_controller(param,value);
237  }
238  break;
239  }
240  }
241  break;
