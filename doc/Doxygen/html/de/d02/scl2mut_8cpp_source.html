121 void run() {
122  std::string s;
123 
124  input->seekg(0, std::ios::end);
125  streampos pos = input->tellg();
126  if (pos > 0)
127  s.reserve(pos);
128  input->seekg(0, std::ios::beg);
129 
130  s.assign((std::istreambuf_iterator<char>(*input)),
131  std::istreambuf_iterator<char>());
132 
133  mutabor::scala_parser::parser scparser(s,inputfile);
134  if (keymapinput) {
135  s.clear();
136  keymapinput->seekg(0, std::ios::end);
137  streampos pos = keymapinput->tellg();
138  if (pos > 0)
139  s.reserve(pos);
140  keymapinput->seekg(0, std::ios::beg);
141 
142  s.assign((std::istreambuf_iterator<char>(*keymapinput)),
143  std::istreambuf_iterator<char>());
144  scparser.load_keymap(s,kbmfile);
145  } else {
146  scparser.make_keymap();
147  }
148 
149  scparser.write_mutabor(*output,options);
150 }
