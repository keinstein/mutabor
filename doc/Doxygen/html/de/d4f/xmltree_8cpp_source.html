365  void xmltree::SetPath(const wxString & path)
366  {
367 
368  /* Warning!!!
369  This function as well as the inner loop have several exit points.
370  The loop step is done using goto.
371  */
372  bool create = create_tree;
373  if (!path) return;
374  wxStringTokenizer tokenizer(path,_T("/"));
375  if (!tokenizer.HasMoreTokens()) return;
376  wxString name = tokenizer.GetNextToken();
377  node_type * node;
378  if (!name) {
379  if (!tokenizer.HasMoreTokens()) {
380  current_node = NULL;
381  return;
382  }
383  name = tokenizer.GetNextToken();
384  node = GetRoot();
385  if (!node) {
386  if (!create) {
387  current_node = NULL;
388  return;
389  }
390  node = new node_type(current_node,
391  wxXML_ELEMENT_NODE,
392  name);
393  SetRoot(node);
394  } else if (node->GetName() != name) {
395  current_node = NULL;
396  return;
397  }
398  current_node = node;
399  name = _T(".");
400  } else if (!GetRoot() || !current_node) return;
401  bool cont = false;
402  do {
403  cont = tokenizer.HasMoreTokens();
404  if (!name || name == _T(".")) {
405  goto loop_step;
406  }
407  if (!current_node) {
408  if (GetRoot()->GetName() == name) {
409  current_node = GetRoot();
410  goto loop_step;
411  }
412  return;
413  }
414  if (name == _T("..")) {
415  if (current_node == GetRoot()) {
416  current_node = NULL;
417  return;
418  }
419  current_node = current_node->GetParent();
420  goto loop_step;
421  }
422  node = current_node->GetChildren();
423  while (node && node -> GetName() != name) {
424  node = node->GetNext();
425  }
426  if (!node) {
427  if (!create) {
428  current_node = NULL;
429  return;
430  }
431  node = new node_type(current_node,
432  wxXML_ELEMENT_NODE,
433  name);
434  }
435 
436  current_node = node;
437  loop_step:
438  if (cont) {
439  name = tokenizer.GetNextToken();
440  }
441  } while (cont);
442  return;
443  }

445  void xmltree::DeleteEntry(const wxString & path) {
446  if (!current_node) return;
447 #if wxCHECK_VERSION(2,9,0)
448  wxXmlAttribute * prop =
449  current_node -> GetAttributes();
450  wxXmlAttribute * last_prop = prop;
451 #else
452  wxXmlProperty * prop =

454  wxXmlProperty * last_prop = prop;
455 #endif
456  while (prop && prop->GetName() != path) {
457  last_prop = prop;
458  prop = prop->GetNext();
459  }
460  if (!prop) {
461  wxXmlNode * node = current_node;
462  SetPath(path);
463  if (current_node) {
464  wxXmlNodeType type = current_node->GetType();
465  switch (type) {
466  case wxXML_ELEMENT_NODE:
467  case wxXML_ATTRIBUTE_NODE:
468  case wxXML_ENTITY_REF_NODE:
469  case wxXML_ENTITY_NODE:
470  case wxXML_PI_NODE:
471  case wxXML_COMMENT_NODE:
472  case wxXML_DOCUMENT_NODE:
473  case wxXML_DOCUMENT_TYPE_NODE:
474  case wxXML_DOCUMENT_FRAG_NODE:
475  case wxXML_NOTATION_NODE:
476  case wxXML_HTML_DOCUMENT_NODE:
477  break;
478  case wxXML_TEXT_NODE:
479  case wxXML_CDATA_SECTION_NODE:
480  delete current_node;
481  }
482  }
483  current_node = node;
484  }
485  if (last_prop != prop) {
486  last_prop->SetNext(prop->GetNext());
