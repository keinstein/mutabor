257 void MutPanel::OnSize(wxSizeEvent& event)
258 {
259  if (GetAutoLayout())
260  Layout();
261 #if wxUSE_CONSTRAINTS
262 #if defined(__WXPM__) && 0
263  else
264  {
265  // Need to properly move child windows under OS/2
266 
267  PSWP pWinSwp = GetSwp();
268 
269  if (pWinSwp->cx == 0 && pWinSwp->cy == 0 && pWinSwp->fl == 0)
270  {
271  // Uninitialized
272 
273  ::WinQueryWindowPos(GetHWND(), pWinSwp);
274  }
275  else
276  {
277  SWP vSwp;
278  int nYDiff;
279 
280  ::WinQueryWindowPos(GetHWND(), &vSwp);
281  nYDiff = pWinSwp->cy - vSwp.cy;
282  MoveChildren(nYDiff);
283  pWinSwp->cx = vSwp.cx;
284  pWinSwp->cy = vSwp.cy;
285  }
286  }
287 #endif
288 #endif // wxUSE_CONSTRAINTS
