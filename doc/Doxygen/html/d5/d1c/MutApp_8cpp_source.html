535  void MutApp::OnMRUFile(wxCommandEvent& event)
536  {
537  int n = event.GetId() - wxID_FILE1; // the index in MRU list
538  wxString filename(document_manager->GetHistoryFile(n));
539  if ( !filename.empty() )
540  {
541  // verify that the file exists before doing anything else
542  if ( wxFile::Exists(filename) )
543  {
544  // try to open it
545  if (!document_manager->CreateDocument(filename, wxDOC_SILENT))
546  {
547  // remove the file from the MRU list. The user should already be notified.
548  document_manager->RemoveFileFromHistory(n);
549 
550  wxLogError(_("The file '%s' couldn't be opened.\nIt has been removed from the most recently used files list."),
551  filename.c_str());
552  } else {
553  wxView * v = document_manager->GetCurrentView();
554  if (v) {
555  wxWindow * w = v->GetFrame();
556  if (w) {
557  w->Raise();
558  }
559  }
560  }
561  }
562  else
563  {
564  // remove the bogus filename from the MRU list and notify the user
565  // about it
566  document_manager->RemoveFileFromHistory(n);
567 
568  wxLogError(_("The file '%s' doesn't exist and couldn't be opened.\nIt has been removed from the most recently used files list."),
569  filename.c_str());
570  }
571  }
572  }
