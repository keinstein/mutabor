195 void configtree::toParent(unsigned int count)
196 {
197  DEBUGLOG(config,_T("going up %d level groups"),count);
198  while (count--) {
199 #ifdef DEBUG
200  mutASSERT(states.size());
201 #endif
202  if (!states.size()) {
203  UNREACHABLEC;
204  return;
205  }
206  state oldstate = states.top();
207  DEBUGLOG(config,_T("going up for %d levels"),oldstate.chdepth);
208  mutASSERT(oldstate.chdepth >= 0);
209  if (oldstate.chdepth) {
210  wxString setpath = _T("..");
211  while (--(oldstate.chdepth))
212  setpath << _T("/..");
213  config -> SetPath(setpath);
214  }
215  states.pop();
216 #ifdef DEBUG
217  mutASSERT(config->GetPath() == oldstate.oldpath);
218 #endif
219  DEBUGLOG(config,
220  _T("current path = '%s', oldstate(oldpath = '%s', chdepth = %d, leafid = %ld, group = '%s')"),
221  config->GetPath().c_str(),
222  oldstate.oldpath.c_str(),
223  oldstate.chdepth, oldstate.leafid,
224  oldstate.group.c_str());
225  }
