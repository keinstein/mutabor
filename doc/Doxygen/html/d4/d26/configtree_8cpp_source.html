200 void configtree::toParent(unsigned int count)
201 {
202  DEBUGLOG (config, "going up %d level groups" ,count);
203  while (count--) {
204 #ifdef DEBUG
205  mutASSERT(states.size());
206 #endif
207  if (!states.size()) {
208  UNREACHABLEC;
209  return;
210  }
211  state oldstate = states.top();
212  DEBUGLOG (config, "going up for %d levels" ,oldstate.chdepth);
213  mutASSERT(oldstate.chdepth >= 0);
214  if (oldstate.chdepth) {
215  wxString setpath = _T("..");
216  while (--(oldstate.chdepth))
217  setpath << _T("/..");
218  config -> SetPath(setpath);
219  }
220  states.pop();
221 #ifdef DEBUG
222  mutASSERT(config->GetPath() == oldstate.oldpath);
223 #endif
224  DEBUGLOG(config,
225  ("current path = '%s', oldstate(oldpath = '%s', chdepth = %d, leafid = %ld, group = '%s')"),
226  config->GetPath().c_str(),
227  oldstate.oldpath.c_str(),
228  oldstate.chdepth, oldstate.leafid,
229  oldstate.group.c_str());
230  }
