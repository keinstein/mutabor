232  template<class iterator>
233  MidiPortOutputProvider & SendSysEx (int channel,
234  iterator from,
235  iterator to) {
236  mutUnused(channel);
237  if (from == to) return * this;
238  if ((*from) & midi::STARTBYTE_MASK) {
239  UNREACHABLEC;
240  return *this;
241  }
242 
243  std::vector<unsigned char> message(to - from + 2);
244  message[0] = midi::SYSEX_START;
245  size_t i = 1;
246  while (from != to) {
247  message[i++] = (*(from++));
248  }
249  message[i] = midi::SYSEX_END;

251  port->sendMessage(message);
252  return *this;
