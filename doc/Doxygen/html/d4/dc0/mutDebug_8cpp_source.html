114 // Default options for AddressSanitizer in various configurations:
115 // malloc_context_size=5 - limit the size of stack traces collected by ASan
116 // for each malloc/free by 5 frames. These stack traces tend to accumulate
117 // very fast in applications using JIT (v8 in Chrome's case), see
118 // https://code.google.com/p/address-sanitizer/issues/detail?id=177
119 // symbolize=1 - enable in-process symbolization.
120 // legacy_pthread_cond=1 - run in the libpthread 2.2.5 compatibility mode to
121 // work around libGL.so using the obsolete API, see
122 // http://crbug.com/341805. This may break if pthread_cond_t objects are
123 // accessed by both instrumented and non-instrumented binaries (e.g. if
124 // they reside in shared memory). This option is going to be deprecated in
125 // upstream AddressSanitizer and must not be used anywhere except the
126 // official builds.
127 // check_printf=1 - check the memory accesses to printf (and other formatted
128 // output routines) arguments.
129 // use_sigaltstack=1 - handle signals on an alternate signal stack. Useful
130 // for stack overflow detection.
131 // strip_path_prefix=/../../ - prefixes up to and including this
132 // substring will be stripped from source file paths in symbolized reports
133 // fast_unwind_on_fatal=1 - use the fast (frame-pointer-based) stack unwinder
134 // to print error reports. V8 doesn't generate debug info for the JIT code,
135 // so the slow unwinder may not work properly.
136 // detect_stack_use_after_return=1 - use fake stack to delay the reuse of
137 // stack allocations and detect stack-use-after-return errors.
138 const char * mutASANoptions =
139  "legacy_pthread_cond=1 "

141  "symbolize=1 "

144  "detect_leaks=1 "
145  "strip_path_prefix=/../../ "
146  "fast_unwind_on_fatal=1 "

163 // Default options for ThreadSanitizer in various configurations:
164 // detect_deadlocks=1 - enable deadlock (lock inversion) detection.
165 // second_deadlock_stack=1 - more verbose deadlock reports.
166 // report_signal_unsafe=0 - do not report async-signal-unsafe functions
167 // called from signal handlers.
168 // report_thread_leaks=0 - do not report unjoined threads at the end of
169 // the program execution.

171 // history_size=7 - make the history buffer proportional to 2^7 (the maximum
