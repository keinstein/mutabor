296  static const WORD MAX_CONSOLE_LINES = 500;

302 
303  // allocate a console for this app
304  AllocConsole();
305 
306  // set the screen buffer to be big enough to let us scroll text
307  GetConsoleScreenBufferInfo(GetStdHandle(STD_OUTPUT_HANDLE),
308  &coninfo);
309  coninfo.dwSize.Y = MAX_CONSOLE_LINES;
310  SetConsoleScreenBufferSize(GetStdHandle(STD_OUTPUT_HANDLE),
311  coninfo.dwSize);
312 
313  // redirect unbuffered STDOUT to the console
314  lStdHandle = (long)GetStdHandle(STD_OUTPUT_HANDLE);
315  hConHandle = _open_osfhandle(lStdHandle, _O_TEXT);
316  fp = _fdopen( hConHandle, "w" );
317  *stdout = *fp;
318  setvbuf( stdout, NULL, _IONBF, 0 );
319 
320  // redirect unbuffered STDIN to the console
321  lStdHandle = (long)GetStdHandle(STD_INPUT_HANDLE);
322  hConHandle = _open_osfhandle(lStdHandle, _O_TEXT);
323 
324  fp = _fdopen( hConHandle, "r" );
325  *stdin = *fp;
326  setvbuf( stdin, NULL, _IONBF, 0 );
327 
328  // redirect unbuffered STDERR to the console
329  lStdHandle = (long)GetStdHandle(STD_ERROR_HANDLE);
330  hConHandle = _open_osfhandle(lStdHandle, _O_TEXT);
331  fp = _fdopen( hConHandle, "w" );

389 void mutAssertFailure(const wxChar *file,
390  int line,
391  const wxChar *func,
392  const wxChar *cond,
393  const wxChar *msg)
394 {
395 #ifdef MUTABOR_CPPUNIT
396  {
397  CPPUNIT_NS::Message mesg( "assertion failed",
398  std::string(wxString::Format(_T("%s:%d:%s\nFunction: %s; Expression: %s"),
399  file,
400  line,
401  (msg?msg:_T("")),
402  func,
403  cond).ToUTF8()));
404 
405  wxString filename(file);
406  CPPUNIT_NS::Asserter::fail(mesg,
407  CPPUNIT_NS::SourceLine(std::string(filename.ToUTF8()), line )
408  );
409  }
410 
411 #else
412  if (!std::clog.good()) MutInitConsole();
413  wxString s = wxString::Format(_("%s:%d: An assert failed in %s().\nCondition: %s\nMessage:%s\nDo you want to call the default assert handler?"),
414  file,line,func,cond,msg);
415  std::clog << (const char *) s.ToUTF8() << std::endl;
416 
417 
418  // use wxWidgets ShowAssertDialog function (it's static so we repeat its code, here.
419 
420  {
421  // this variable can be set to true to suppress "assert failure" messages
422  static bool s_bNoAsserts = false;
423 
424  wxString message;
425  message.reserve(2048);
426 
427  // here, we change format to the gcc form
428  message.Printf(wxT("%s:%d: assert \"%s\" failed"), file, line, cond);
429 
430  // add the function name, if any
431  if ( func && *func )
432  message << _T(" in ") << func << _T("()");
433 
434  // and the message itself
435  if ( msg )
436  {
437  message << _T(": ") << msg;
438  }
439  else // no message given
440  {
441  message << _T('.');
442  }
443 
444 #if wxUSE_THREADS
445  // if we are not in the main thread, output the assert directly and trap
446  // since dialogs cannot be displayed
447  if ( !wxThread::IsMain() )
448  {
449  message += wxT(" [in child thread]");
450 
451 #if defined(__WXMSW__) && !defined(__WXMICROWIN__)
452  message << wxT("\r\n");
453  OutputDebugString(message );
454 #else
455  // send to stderr
456  wxFprintf(stderr, wxT("%s\n"), message.c_str());
457  fflush(stderr);
458 #endif
459  // He-e-e-e-elp!! we're asserting in a child thread
460  wxTrap();
461  }
462  else
463 #endif // wxUSE_THREADS
464 
465  if ( !s_bNoAsserts )
466  {
467  // send it to the normal log destination
468  wxLogDebug(_T("%s"), message.c_str());
469 
470 #ifdef __WXDEBUG__
471  if ( wxGetApp().GetTraits() )
472  {
473  // delegate showing assert dialog (if possible) to that class
474  s_bNoAsserts = wxGetApp().GetTraits()->ShowAssertDialog(message);
475  }
476  else // no traits object
477 #endif
478  {
479  // fall back to the function of last resort
480  s_bNoAsserts = DoShowAssertDialog(message);
481  }
482  }
483  }
484 #endif
485 }
