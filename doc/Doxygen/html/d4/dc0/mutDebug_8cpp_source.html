150  static const WORD MAX_CONSOLE_LINES = 500;
151 
152  int hConHandle;
153  long lStdHandle;
154  CONSOLE_SCREEN_BUFFER_INFO coninfo;
155  FILE *fp;
156 
157  // allocate a console for this app
158  AllocConsole();
159 
160  // set the screen buffer to be big enough to let us scroll text
161  GetConsoleScreenBufferInfo(GetStdHandle(STD_OUTPUT_HANDLE),
162  &coninfo);
163  coninfo.dwSize.Y = MAX_CONSOLE_LINES;
164  SetConsoleScreenBufferSize(GetStdHandle(STD_OUTPUT_HANDLE),
165  coninfo.dwSize);
166 
167  // redirect unbuffered STDOUT to the console
168  lStdHandle = (long)GetStdHandle(STD_OUTPUT_HANDLE);
169  hConHandle = _open_osfhandle(lStdHandle, _O_TEXT);
170  fp = _fdopen( hConHandle, "w" );
171  *stdout = *fp;
172  setvbuf( stdout, NULL, _IONBF, 0 );
173 
174  // redirect unbuffered STDIN to the console
175  lStdHandle = (long)GetStdHandle(STD_INPUT_HANDLE);
176  hConHandle = _open_osfhandle(lStdHandle, _O_TEXT);
177 
178  fp = _fdopen( hConHandle, "r" );
179  *stdin = *fp;
180  setvbuf( stdin, NULL, _IONBF, 0 );
181 
182  // redirect unbuffered STDERR to the console
183  lStdHandle = (long)GetStdHandle(STD_ERROR_HANDLE);
184  hConHandle = _open_osfhandle(lStdHandle, _O_TEXT);
185  fp = _fdopen( hConHandle, "w" );
186  *stderr = *fp;
187  setvbuf( stderr, NULL, _IONBF, 0 );
188 
189  // make cout, wcout, cin, wcin, wcerr, cerr, wclog and clog

243 void mutAssertFailure(const wxChar *file,
244  int line,
245  const wxChar *func,
246  const wxChar *cond,
247  const wxChar *msg)
248 {
249 #ifdef MUTABOR_CPPUNIT
250  {
251  CPPUNIT_NS::Message mesg( "assertion failed",
252  std::string(wxString::Format(_T("%s:%d:%s\nFunction: %s; Expression: %s"),
253  file,
254  line,
255  (msg?msg:_T("")),
256  func,
257  cond).ToUTF8()));
258 
259  wxString filename(file);
260  CPPUNIT_NS::Asserter::fail(mesg,
261  CPPUNIT_NS::SourceLine(std::string(filename.ToUTF8()), line )
262  );
263  }
264 
265 #else
266  if (!std::clog.good()) MutInitConsole();
267  wxString s = wxString::Format(_("%s:%d: An assert failed in %s().\nCondition: %s\nMessage:%s\nDo you want to call the default assert handler?"),
268  file,line,func,cond,msg);
269  std::clog << (const char *) s.ToUTF8() << std::endl;
270 
271 
272  // use wxWidgets ShowAssertDialog function (it's static so we repeat its code, here.
273 
274  {
275  // this variable can be set to true to suppress "assert failure" messages
276  static bool s_bNoAsserts = false;
277 
278  wxString message;
279  message.reserve(2048);
280 
281  // here, we change format to the gcc form
282  message.Printf(wxT("%s:%d: assert \"%s\" failed"), file, line, cond);
283 
284  // add the function name, if any
285  if ( func && *func )
286  message << _T(" in ") << func << _T("()");
287 
288  // and the message itself
289  if ( msg )
290  {
291  message << _T(": ") << msg;
292  }
293  else // no message given
294  {
295  message << _T('.');
296  }
297 
298 #if wxUSE_THREADS
299  // if we are not in the main thread, output the assert directly and trap
300  // since dialogs cannot be displayed
301  if ( !wxThread::IsMain() )
302  {
303  message += wxT(" [in child thread]");
304 
305 #if defined(__WXMSW__) && !defined(__WXMICROWIN__)
306  message << wxT("\r\n");
307  OutputDebugString(message );
308 #else
309  // send to stderr
310  wxFprintf(stderr, wxT("%s\n"), message.c_str());
311  fflush(stderr);
312 #endif
313  // He-e-e-e-elp!! we're asserting in a child thread
314  wxTrap();
315  }
316  else
317 #endif // wxUSE_THREADS
318 
319  if ( !s_bNoAsserts )
320  {
321  // send it to the normal log destination
322  wxLogDebug(_T("%s"), message.c_str());
323 
324 #ifdef __WXDEBUG__
325  if ( wxGetApp().GetTraits() )
326  {
327  // delegate showing assert dialog (if possible) to that class
328  s_bNoAsserts = wxGetApp().GetTraits()->ShowAssertDialog(message);
329  }
330  else // no traits object
331 #endif
332  {
333  // fall back to the function of last resort
334  s_bNoAsserts = DoShowAssertDialog(message);
335  }
336  }
337  }
338 #endif
339 }
