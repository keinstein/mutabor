67 muConvAuto::BOMType muConvAuto::DetectBOM(const char *src, size_t srcLen)
68 {
69  DEBUGLOGTYPE(other,muConvAuto,_T("Detecting BOM at length %d"),(int)srcLen);
70  if (srcLen == wxNO_LEN) { // zero terminated string
71  for (srcLen = 0; src[srcLen] && srcLen <= 4; srcLen++);
72  }
73  if ( srcLen < 2 ) {
74  // minimal BOM is 2 bytes so bail out immediately and simplify the code
75  // below which wouldn't need to check for length for UTF-16 cases
76  return BOM_None;
77  }
78 
79  // examine the buffer for BOM presence
80  //
81  // see http://www.unicode.org/faq/utf_bom.html#BOM
82  switch ( *(src++) ) {
83  case '\0':
84  // could only be big endian UTF-32 (00 00 FE FF)
85  if ( srcLen >= 4 &&
86  src[0] == '\0' &&
87  src[1] == '\xfe' &&
88  src[2] == '\xff' ) {
89  return BOM_UTF32BE;
90  }
91  break;
92 
93  case '\xfe':
94  // could only be big endian UTF-16 (FE FF)
95  if ( *src++ == '\xff' ) {
96  return BOM_UTF16BE;
97  }
98  break;
99 
100  case '\xff':
101  // could be either little endian UTF-16 or UTF-32, both start
102  // with FF FE
103  if ( *(src++) == '\xfe' ) {
104  return srcLen >= 4 && src[0] == '\0' && src[1] == '\0'
105  ? BOM_UTF32LE
106  : BOM_UTF16LE;
107  }
108  break;
109 
110  case '\xef':
111  // is this UTF-8 BOM (EF BB BF)?
112  if ( srcLen >= 3 && src[0] == '\xbb' && src[1] == '\xbf' ) {
113  return BOM_UTF8;
114  }
115  break;
116  }
117 
118  return BOM_None;
119 }

121 void muConvAuto::InitFromBOM(BOMType bomType)
122 {
123  TRACEC;
124  m_consumedBOM = false;
125 
126  switch ( bomType ) {
127  case BOM_UTF32BE:
128  m_conv = new wxMBConvUTF32BE;
129  m_ownsConv = true;
130  break;
131 
132  case BOM_UTF32LE:
133  m_conv = new wxMBConvUTF32LE;
134  m_ownsConv = true;
135  break;
136 
137  case BOM_UTF16BE:
138  m_conv = new wxMBConvUTF16BE;
139  m_ownsConv = true;
140  break;
141 
142  case BOM_UTF16LE:
143  m_conv = new wxMBConvUTF16LE;
144  m_ownsConv = true;
145  break;
146 
147  case BOM_UTF8:
148  m_conv = &wxConvUTF8;
149  m_ownsConv = false;
150  break;
151 
152  default:
153  wxFAIL_MSG( _T("unexpected BOM type") );
154  // fall through: still need to create something
155 
156  case BOM_None:
157  InitWithDefault();
158  m_consumedBOM = true; // as there is nothing to consume
159  }
160 }

162 void muConvAuto::SkipBOM(const char **src, size_t *len) const
163 {
164  TRACEC;
165  int ofs;
166  switch ( m_bomType ) {
167  case BOM_UTF32BE:
168  case BOM_UTF32LE:
169  ofs = 4;
170  break;
171 
172  case BOM_UTF16BE:
173  case BOM_UTF16LE:
174  ofs = 2;
175  break;
176 
177  case BOM_UTF8:
178  ofs = 3;
179  break;
180 
181  default:
182  wxFAIL_MSG( _T("unexpected BOM type") );
183  // fall through: still need to create something
184 
185  case BOM_None:
186  ofs = 0;
187  }
188 
189  *src += ofs;
190  if ( *len != wxNO_LEN )
191  *len -= ofs;
192 }

203 size_t muConvAuto::ToWChar(wchar_t *dst, size_t dstLen,
204  const char *src, size_t srcLen) const
205 {
206  TRACEC;
207  // we check BOM and create the appropriate conversion the first time we're
208  // called but we also need to ensure that the BOM is skipped not only
209  // during this initial call but also during the first call with non-NULL
210  // dst as typically we're first called with NULL dst to calculate the
211  // needed buffer size
212  muConvAuto *self = wx_const_cast(muConvAuto *, this);
213  if ( !m_conv ) {
214  self->InitFromInput(&src, &srcLen);
215  if ( dst )
216  self->m_consumedBOM = true;
217  }
218 
219  if ( !m_consumedBOM && dst ) {
220  self->m_consumedBOM = true;
221  SkipBOM(&src, &srcLen);
222  }
223 
224  size_t result = m_conv->ToWChar(dst, dstLen, src, srcLen);
225  if (result != wxCONV_FAILED) return result;
226 
227  self->m_conv = m_fallback; // save for further use.
228  self->m_ownsConv = false; // if we own fallback, it will be destroyed as fallback.
229  return m_conv->ToWChar(dst, dstLen, src, srcLen);
230 }
