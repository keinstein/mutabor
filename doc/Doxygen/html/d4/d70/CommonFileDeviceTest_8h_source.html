85  mutint64 newtime = mutabor::CurrentTime.Get();
86  mutint64 tl = newtime;
87 // tl = tl - (lasttime + (mutint64)(i*(i+1))/2);
88  tl = tl - lasttime;
89  mutint64 deviation = tl - (mutint64) i*1000;
90  if (max < deviation) max = deviation;
91  if (min > deviation || min == 0) min = deviation;
92  DEBUGLOG(timer,_T("deviation = %ld, min = %ld, max = %ld"), (long)deviation, (long)min, (long)max);
93  if (deviation > 10000) {
94  std::cerr << "Too slow: (" << i << "^2 + " << i << ") / 2 = " << (i*(i+1))/2
95  << " Runtime: " << tl << "μs" << std::endl;
96  CPPUNIT_ASSERT(deviation <= 10000);
97  }
98  if (deviation < -10000) {
99  std::cerr << "Too fast: (" << i << "^2 + " << i << ") / 2 = " << (i*(i+1))/2

101  CPPUNIT_ASSERT(deviation >= -10000);
102  }
103 
104  lasttime = newtime;
