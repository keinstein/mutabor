97  mutabor::microseconds PrepareNextEvent() {
98  mutabor::ScopedLock<> lock(timelock);
99  /* race condition: Mode should be DevicePlay, but it
100  may be any error free state as the device might
101  have been paused or stopped already. */
102  CPPUNIT_ASSERT(Mode == mutabor::DevicePlay ||
103  Mode == mutabor::DevicePause ||
104  Mode == mutabor::DeviceStop);
105  time_point newtime = clocktype::now();
106  milliseconds tl = boost::chrono::duration_cast<milliseconds>(newtime - lasttime);
107 // tl = tl - (lasttime + (mutint64)(i*(i+1))/2);
108  milliseconds deviation = tl - i;
109  if (max < deviation) max = deviation;
110  if (min > deviation || min == milliseconds(0)) min = deviation;
111  DEBUGLOG (timer, "deviation = %ld, min = %ld, max = %ld" , deviation, min, max);
112  if (deviation > milliseconds(10)) {
113  std::cerr << "Too slow: (" << i << "^2 + " << i << ") / 2 = " << (i*(i.count()+1))/2
114  << " Runtime: " << tl << "ms" << std::endl;
115  CPPUNIT_ASSERT(deviation <= milliseconds(10));
116  }
117  if (deviation < milliseconds(-10)) {
118  std::cerr << "Too fast: (" << i << "^2 + " << i << ") / 2 = " << (i*(i.count()+1))/2
119  << " Runtime: " << tl << "μs" << std::endl;
120  CPPUNIT_ASSERT(deviation >= milliseconds(-10));
121  }
122 
123  lasttime = newtime;
124  if (++i<milliseconds(100)) {
125  DEBUGLOG (timer, "Returning %ld μs" ,(i));
126  DEBUGLOG (timer, "Time spent in function: %ldns" , (clocktype::now()-newtime));
127  return (i);
128  }
129  return NO_DELTA();
130  }
