105 int MisFile::Load(char *fileName)
106 {
107  ifstream is(fileName, ios::in | ios::binary);
108  if ( is.bad() )
109  return is.bad();
110  // header mark
111  char Header[5];
112  is.get(Header, 5);
113  DWORD l = ReadLength(is);
114  BYTE a, b;
115  // file type
116  is >> a >> Type;
117  // number of tracks
118  is >> a >> b;
119  int NTrack = (((int)a) << 8) + b;
120  // speed info
121  is >> a >> b;
122  Speed = (((int)a) << 8) + b;
123  // rest of header
124  for (DWORD i = 6; i < l && !is.eof() && !is.bad(); i++ )
125  {
126  is.get(a);
127  Add(a);
128  }
129  int res = is.bad();
130  // read tracks
131  MisTrack **CurrentTrack = &Tracks;
132  for (i = 0; i < NTrack && !res; i++ )
133  {
134  MisTrack *Track = new MisTrack(this, i);
135  res = Track->Load(is);
136  *CurrentTrack = Track;
137  CurrentTrack = &Track->Next;
138  }
139  if ( !res ) is.close();
140  return res;
141 }

467 // reads the next midi event
468 DWORD MisTrack::ReadMidi(DWORD &pos)
469 {
470  DWORD a = Get(pos++);
471  if ( a & 0x80 )
472  StatusByte = a;
473  else
474  {
475  pos--;
476  a = StatusByte;
477  }
478  if ( StatusByte < 0xF0 ) // normaler midi code
479  {
480  int l = lMidiCode[(StatusByte >> 4) & 07];
481  int shift = 8;
482  for (int i = 1; i < l; i++)
483  {
484  a += (DWORD)Get(pos++) << shift;
485  shift += 8;
486  }
487  }
488  else if ( StatusByte == 0xF0 || StatusByte == 0xF7 ) // SysEx I, SysEx II
489  {
490  EventLength = ReadDeltaTime(pos); // length
491  EventPos = pos; // position of Data
492  pos += EventLength;
493  }
494  else if ( StatusByte == 0xFF ) // meta event
495  {
496  a += Get(pos++) << 8; // event number
497  EventLength = ReadDeltaTime(pos); // length
498  EventPos = pos; // position of Data
499  pos += EventLength;
500  }
501  else if ( StatusByte == 0xF2 ) // song position pointer
502  {
503  a += Get(pos++) << 8;
504  a += Get(pos++) << 16;
505  }
506  else if ( StatusByte == 0xF3 ) // song select
507  a += Get(pos++) << 8;
508  return a;
509 }
