453  inline event_class * create_meta_event(const std::vector<unsigned char> &code,
454  size_t unique_id) {
455  if (code.size() < 2) {
456  return new meta_event(unique_id, code);
457  }
458 
459 
460  switch (code[1]) {
461  case midi::META_SEQUENCE_NUMBER: {
462  if (code.size() >= 4) {
463  return new sequence_number_event(unique_id,
464  (code[2] << 8) & code[3]);
465  }
466  }
467  break;
468  case midi::META_EVENT_TEXT: {
469  if (code.size() >= 3)
470  return new text_event(unique_id,
471  &(code[2]),
472  code.size()-2);
473  }
474  break;
475  case midi::META_COPYRIGHT_NOTICE: {
476  if (code.size() >= 3)
477  return new copyright_event(unique_id,
478  &(code[2]),
479  code.size()-2);
480  }
481  break;
482  case midi::META_SEQUENCE_NAME: {
483  if (code.size() >= 3) {
484  // Sequcence and track are something different in MIDI.
485  // Nevertheless they share the same id.
486  if (false)
487  return new sequence_name_event(unique_id,
488  &(code[2]),
489  code.size()-2);
490  else
491  return new track_name_event(unique_id,
492  &(code[2]),
493  code.size()-2);
494  }
495  }
496  break;
497 
498  case midi::META_INSTRUMENT_NAME: {
499  if (code.size() >= 3)
500  return new instrument_name_event(unique_id,
501  &(code[2]),
502  code.size()-2);
503  }
504  break;
505  case midi::META_LYRIC: {
506  if (code.size() >= 3)
507  return new lyric_event(unique_id,
508  &(code[2]),
509  code.size()-2);
510  }
511  break;
512  case midi::META_MARKER: {
513  if (code.size() >= 3)
514  return new marker_event(unique_id,
515  &(code[2]),
516  code.size()-2);
517  }
518  break;
519  case midi::META_CUE_POINT: {
520  if (code.size() >= 3)
521  return new cue_point_event(unique_id,
522  &(code[2]),
523  code.size()-2);
524  }
525  break;
526  case midi::META_CHANNEL_PREFIX: {
527  if (code.size() >= 3)
528  return new channel_select_event(unique_id,
529  (code[2]));
530  }
531  break;
532  case midi::META_END_OF_TRACK: {
533  return new end_track_event(unique_id);
534  }
535  break;
536  case midi::META_SET_TEMPO: {
537  if (code.size() >= 5)
538  return new set_tempo_event(unique_id,
539  (code[2] << 16) | (code[3] << 8) | code[4]);
540  }
541  break;
542  case midi::META_SMPTE_OFFSET: {
543  if (code.size() >= 7)
544  return new smpte_event(unique_id,
545  code[2],
546  code[3],
547  code[4],
548  code[5],
549  code[6]);
550  }
551  break;
552  case midi::META_TIME_SIGNATURE: {
553  if (code.size() >= 6)
554  return new time_signature_event(unique_id,
555  code[2],
556  1 << code[3],
557  code[4],
558  code[5]);
559  }
560  break;
561  case midi::META_KEY_SIGNATURE: {
562  if (code.size() >= 3)
563  return new key_signature_event(unique_id,
564  code[2],
565  ! code[3]);
566  }
567  break;
568  case midi::META_SEQUENCER_SPECIFIC:
569  return new sequencer_specific_meta_event(unique_id, code);
570  default:
571  return new meta_event(unique_id, code);
572  }
573  return new meta_event(unique_id, code);
574  }

576  inline event_class * create_system_event(const std::vector<unsigned char> &code,
577  size_t unique_id) {
578  switch (code[0]) {
579  case midi::SYSEX_START:
580  if (code.size() < 2) {
581  std::vector<unsigned char> tmp;
582  return new sysex_event(unique_id);
583  }
584  return new sysex_event (unique_id,code);
585  case midi::QUARTER_FRAME:
586  if (code.size() > 1)
587  return new quarter_frame_event(unique_id,
588  (code[1] & 0x70) >> 4,
589  code[1] & 0x0F);
590  break;
591  case midi::SONG_POSITION:
592  if (code.size() > 2)
593  return new song_position_event(unique_id,
594  code[1]
595  | (code[2] << 7));
596  break;
597  case midi::SONG_SELECT:
598  if (code.size() > 1)
599  return new song_select_event(unique_id,
600  code[1]);
601  break;
602  case midi::SYSTEM_UNDEFINED1:
603  return new system_common_midi_undef1_event(unique_id);
604  case midi::SYSTEM_UNDEFINED2:
605  return new system_common_midi_undef2_event(unique_id);
606  case midi::TUNE_REQUEST:
607  return new tune_request_event(unique_id);
608  case midi::SYSEX_END:
609  return new sysex_end_event(unique_id);
610  case midi::CLOCK:
611  return new system_realtime_clock_event(unique_id);
612  case midi::TICK:
613  return new system_realtime_tick_event(unique_id);
614  case midi::START_PLAY:
615  return new start_play_event(unique_id);
616  case midi::CONTINUE_PLAY:
617  return new continue_play_event(unique_id);
618  case midi::STOP_PLAY:
619  return new stop_play_event(unique_id);
620  case midi::REALTIME_UNDEFINED:
621  return new system_realtime_midi_undef2(unique_id);
622  case midi::ACTIVE_SENSE:
623  return new active_sense_event(unique_id);
624  case midi::META:
625  if (code.size() > 1)
626  return create_meta_event(code,unique_id);
627  else
628  return new reset_event(unique_id);
629  }
630  return NULL;
631 
632  }

634  inline event_class * create_event(const std::vector<unsigned char> &code,
635  size_t unique_id) {
636  unsigned char midiStatus = code[0] & 0xF0;
637  /* invalid midi messsages are handled by the default branch */
638  switch (midiStatus) {
639  case midi::SYSTEM:
640  return create_system_event(code,unique_id);
641  case midi::CHANNEL_PRESSURE:
642  if (code.size() > 1)
643  return new controller_event(code[0] & 0x0f,
644  unique_id,
645  midi::CHANNEL_PRESSURE_VAL,
646  code[1]);
647  FALLTHROUGH;
648  case midi::KEY_PRESSURE:
649  if (code.size() > 2)
650  return new key_pressure_event(code[0] & 0x0f,
651  unique_id,
652  code[1],
653  code[2]);
654  FALLTHROUGH;
655  case midi::CONTROLLER:
656  if (code.size() > 2)
657  return new controller_event(code[0] & 0x0f,
658  unique_id,
659  code[1],
660  code[2]);
661  FALLTHROUGH;
662  default:
663  return new event_class;
664  }
665  }
