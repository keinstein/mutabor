452  inline event_class * create_meta_event(const std::vector<unsigned char> * code,
453  size_t unique_id) {
454  if (!code) {
455  std::vector<unsigned char> tmp;
456  return new meta_event(unique_id, tmp);
457  }
458  if (code->size() < 2) {
459  return new meta_event(unique_id, *code);
460  }
461 
462 
463  switch (code->at(1)) {
464  case midi::META_SEQUENCE_NUMBER: {
465  if (code->size() >= 4) {
466  return new sequence_number_event(unique_id,
467  (code->at(2) << 8) & code->at(3));
468  }
469  }
470  break;
471  case midi::META_EVENT_TEXT: {
472  if (code->size() >= 3)
473  return new text_event(unique_id,
474  &(code->at(2)),
475  code->size()-2);
476  }
477  break;
478  case midi::META_COPYRIGHT_NOTICE: {
479  if (code->size() >= 3)
480  return new copyright_event(unique_id,
481  &(code->at(2)),
482  code->size()-2);
483  }
484  break;
485  case midi::META_SEQUENCE_NAME: {
486  if (code->size() >= 3) {
487  // Sequcence and track are something different in MIDI.
488  // Nevertheless they share the same id.
489  if (false)
490  return new sequence_name_event(unique_id,
491  &(code->at(2)),
492  code->size()-2);
493  else
494  return new track_name_event(unique_id,
495  &(code->at(2)),
496  code->size()-2);
497  }
498  }
499  break;
500 
501  case midi::META_INSTRUMENT_NAME: {
502  if (code->size() >= 3)
503  return new instrument_name_event(unique_id,
504  &(code->at(2)),
505  code->size()-2);
506  }
507  break;
508  case midi::META_LYRIC: {
509  if (code->size() >= 3)
510  return new lyric_event(unique_id,
511  &(code->at(2)),
512  code->size()-2);
513  }
514  break;
515  case midi::META_MARKER: {
516  if (code->size() >= 3)
517  return new marker_event(unique_id,
518  &(code->at(2)),
519  code->size()-2);
520  }
521  break;
522  case midi::META_CUE_POINT: {
523  if (code->size() >= 3)
524  return new cue_point_event(unique_id,
525  &(code->at(2)),
526  code->size()-2);
527  }
528  break;
529  case midi::META_CHANNEL_PREFIX: {
530  if (code->size() >= 3)
531  return new channel_select_event(unique_id,
532  (code->at(2)));
533  }
534  break;
535  case midi::META_END_OF_TRACK: {
536  return new end_track_event(unique_id);
537  }
538  break;
539  case midi::META_SET_TEMPO: {
540  if (code->size() >= 5)
541  return new set_tempo_event(unique_id,
542  (code->at(2) << 16) | (code->at(3) << 8) | code->at(4));
543  }
544  break;
545  case midi::META_SMPTE_OFFSET: {
546  if (code->size() >= 7)
547  return new smpte_event(unique_id,
548  code->at(2),
549  code->at(3),
550  code->at(4),
551  code->at(5),
552  code->at(6));
553  }
554  break;
555  case midi::META_TIME_SIGNATURE: {
556  if (code->size() >= 6)
557  return new time_signature_event(unique_id,
558  code->at(2),
559  1 << code->at(3),
560  code->at(4),
561  code->at(5));
562  }
563  break;
564  case midi::META_KEY_SIGNATURE: {
565  if (code->size() >= 3)
566  return new key_signature_event(unique_id,
567  code->at(2),
568  ! code->at(3));
569  }
570  break;
571  case midi::META_SEQUENCER_SPECIFIC:
572  return new sequencer_specific_meta_event(unique_id, *code);
573  default:
574  return new meta_event(unique_id, *code);
575  }
576  return new meta_event(unique_id, *code);
577  }

579  inline event_class * create_system_event(const std::vector<unsigned char> * code,
580  size_t unique_id) {
581  mutASSERT(code);
582  switch (code -> at(0)) {
583  case midi::SYSEX_START:
584  if (code->size() < 2) {
585  std::vector<unsigned char> tmp;
586  return new sysex_event(unique_id);
587  }
588  return new sysex_event (unique_id,*code);
589  case midi::QUARTER_FRAME:
590  if (code->size() > 1)
591  return new quarter_frame_event(unique_id,
592  (code->at(1) & 0x70) >> 4,
593  code->at(1) & 0x0F);
594  break;
595  case midi::SONG_POSITION:
596  if (code->size() > 2)
597  return new song_position_event(unique_id,
598  code->at(1)
599  | (code -> at(2) << 7));
600  break;
601  case midi::SONG_SELECT:
602  if (code->size() > 1)
603  return new song_select_event(unique_id,
604  code->at(1));
605  break;
606  case midi::SYSTEM_UNDEFINED1:
607  return new system_common_midi_undef1_event(unique_id);
608  case midi::SYSTEM_UNDEFINED2:
609  return new system_common_midi_undef2_event(unique_id);
610  case midi::TUNE_REQUEST:
611  return new tune_request_event(unique_id);
612  case midi::SYSEX_END:
613  return new sysex_end_event(unique_id);
614  case midi::CLOCK:
615  return new system_realtime_clock_event(unique_id);
616  case midi::TICK:
617  return new system_realtime_tick_event(unique_id);
618  case midi::START_PLAY:
619  return new start_play_event(unique_id);
620  case midi::CONTINUE_PLAY:
621  return new continue_play_event(unique_id);
622  case midi::STOP_PLAY:
623  return new stop_play_event(unique_id);
624  case midi::REALTIME_UNDEFINED:
625  return new system_realtime_midi_undef2(unique_id);
626  case midi::ACTIVE_SENSE:
627  return new active_sense_event(unique_id);
628  case midi::META:
629  if (code->size() > 1)
630  return create_meta_event(code,unique_id);
631  else
632  return new reset_event(unique_id);
633  }
634  return NULL;
635 
636  }

638  inline event_class * create_event(const std::vector<unsigned char> * code,
639  size_t unique_id) {
640  mutASSERT(code);
641  unsigned char midiStatus = code -> at(0) & 0xF0;
642  switch (midiStatus) {
643  case midi::KEY_PRESSURE:
644  if (code->size() > 2)
645  return new key_pressure_event(code->at(0) & 0x0f,
646  unique_id,
647  code->at(1),
648  code->at(2));
649  case midi::CONTROLLER:
650  if (code->size() > 2)
651  return new controller_event(code->at(0) & 0x0f,
652  unique_id,
653  code->at(1),
654  code->at(2));
655  case midi::CHANNEL_PRESSURE:
656  if (code->size() > 1)
657  return new controller_event(code->at(0) & 0x0f,
658  unique_id,
659  midi::CHANNEL_PRESSURE_VAL,
660  code->at(1));
661  case midi::SYSTEM:
662  return create_system_event(code,unique_id);
663  default:
664  return new event_class;
665  }
666  }
