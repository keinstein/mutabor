99  {
100  size_t l = p.length();
101  start:
102  if ( i >= l )
103  return false;
104  while ( i < l && find_in_string(" \t\n\r",p[i]))
105  i++;
106  if ( i >= l )
107  return false;
108  size_t i1 = i;
109  int i2 = p.find(("\n"), i1);
110  if ( i2 == -1 ) {
111  i = l;
112  s = p.substr(i1);
113  } else {
114  s = p.substr(i1, i2-i1);
115  i = (size_t) i2;
116  }
117 
118  boost::trim(s,nolocale);
119  if ( s.empty() || s[0] == '#')
120  goto start;
121 
122  return true;
