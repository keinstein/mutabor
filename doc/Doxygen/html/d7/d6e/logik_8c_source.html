201

203

206  fatal_error (51, lauf->name);
207  }
208  }
209

214  /* Einst. in Logik */
215  }

218  fatal_error(62,lauf->name, lauf->einstimmungs_name);
219  }
220  }
221  }
222  }
223 
224  for (anweisungs_lauf = lauf->anweisungsliste,
