419 void MutRouteWnd::CmRouteLoad(wxCommandEvent& event)
420 {
421  // it's our task
422  event.Skip(false);
423 
424  MutFileDataType filedata = FileNameDialog(wxGetApp().GetTopWindow(),
425  event.GetId());
426 
427 
428  switch (filedata.type) {
429  case MutFileDataType::Canceled:
430  return;
431  case MutFileDataType::TextRoute:
432  case MutFileDataType::UTF8TextRoute:
433  case MutFileDataType::XMLRoute1:
434  break;
435  case MutFileDataType::Unknown:
436  case MutFileDataType::LogicSource:
437  UNREACHABLE;
438  return;
439  }
440  if (!filedata.name.IsOk()) {
441  wxLogError(_("The string '%s' is not a valid file name."),
442  filedata.name.GetFullPath().c_str());
443  return;
444  }
445 
446  MurFileData = filedata;
447 
448 
449  switch (MurFileData.type) {
450  case MutFileDataType::TextRoute:
451  case MutFileDataType::UTF8TextRoute: {
452  wxFFile file(MurFileData.name.GetFullPath());
453  if (!file.IsOpened()) {
454  wxLogError(_("File '%s' couldn't be loaded"),MurFileData.name.GetFullPath().c_str());
455  return;
456  }
457  wxString text;
458  if (file.ReadAll(&text, wxConvUTF8)) {
459  MurFileData.type=MutFileDataType::UTF8TextRoute;
460  } else if (file.ReadAll(&text, wxConvISO8859_1)) {
461  MurFileData.type=MutFileDataType::TextRoute;
462  } else {
463  wxLogError(_("File '%s' could be opened, but not loaded."),
464  MurFileData.name.GetFullPath().c_str());
465  return;
466  }
467  compat30::LoadRoutes((const char *)text.ToUTF8());
468  }
469  break;
470  case MutFileDataType::XMLRoute1: {
471  xmltree document(_T("mutabor-routes"),false);
472  if (!document.Load(MurFileData.name.GetFullPath())) {
473  wxLogError(_("File '%s' couldn't be loaded"),MurFileData.name.GetFullPath().c_str());
474  return;
475  }
476 
477  RouteClass::ClearRouteList();
478  InputDeviceClass::ClearDeviceList();
479  BoxClass::ClearBoxList();
480  OutputDeviceClass::ClearDeviceList();
481 
482  DebugCheckRoutes();
483 
484  mutabor::RouteFactory::LoadRoutes(document);
485  }
486  break;
487  case MutFileDataType::Canceled:
488  case MutFileDataType::Unknown:
489  case MutFileDataType::LogicSource:
490  UNREACHABLE;
491  return;
492  }
493 
494  // Activate the Route window
495  wxGetApp().CmRoutes(event);
496  MutRouteWnd * routewnd=NULL;
497  MutFrame * frame =
498  dynamic_cast<MutFrame*>(wxWindow::FindWindowById(WK_ROUTE));
499  if (frame) {
500  wxWindowList & list = frame->GetChildren();
501  for (wxWindowList::iterator i = list.begin();i!= list.end();i++) {
502  routewnd = dynamic_cast<MutRouteWnd *> (*i);
503  if (routewnd) break;
504  }
505  }
506  mutASSERT(routewnd);
507 
508  if (routewnd) {
509  routewnd->InitDevices();
510  routewnd->Layout();
511  routewnd->FitInside();
512  routewnd->Refresh();
513  }
514 
515  DebugCheckRoutes();
516 
517  wxCommandEvent ev(wxEVT_COMMAND_MENU_SELECTED,CM_ROUTES);
518  wxGetApp().ProcessEvent(ev);
519 }

521 void MutRouteWnd::CmRouteSave(wxCommandEvent& event)
522 {
523  TRACET(MutFrame);
524  event.Skip(false);
525 
526  switch (MurFileData.type) {
527  case MutFileDataType::Canceled:
528  return;
529  case MutFileDataType::Unknown:
530  CmRouteSaveAs(event);
531  return;
532  case MutFileDataType::TextRoute:
533  case MutFileDataType::UTF8TextRoute:
534  case MutFileDataType::XMLRoute1:
535  if (!MurFileData.name.IsOk() ) {
536  CmRouteSaveAs(event);
537  return;
538  }
539  break;
540  case MutFileDataType::LogicSource:
541  UNREACHABLE;
542  return;
543  }
544 
545  /* not applicable, since the file might not yet exsist
546  if (!MurFileData.name.IsFileWritable()) {
547  wxLogError(_("Cannot write to routes file '%s'."),
548  MurFileData.name.GetFullPath().c_str());
549  return;
550  }
551  */
552 
553  switch (MurFileData.type) {
554  case MutFileDataType::TextRoute:
555  case MutFileDataType::UTF8TextRoute: {
556  wxFFile file(MurFileData.name.GetFullPath(), _T("w"));
557 
558  if (!file.IsOpened()) {
559  wxLogError(_("Cannot open routes file '%s' for writing."),
560  MurFileData.name.GetFullPath().c_str());
561  return;
562  }
563 
564  std::string RouteConfig;
565 
566  compat30::SaveRoutes(RouteConfig);
567 
568  wxMBConv &converter = MurFileData.type == MutFileDataType::UTF8TextRoute ?
569  (wxMBConv&)wxConvUTF8 : (wxMBConv&)wxConvISO8859_1;
570 
571  if (file.Write(_T("# Mutabor 3.x routes configuration\n"),
572  converter))
573  if (file.Write(RouteConfig, converter)) {
574  file.Close();
575  return;
576  }
577 
578  wxLogError(_("Error writing file '%s'."),
579  MurFileData.name.GetFullPath().c_str());
580  file.Close();
581  }
582  break;
583  case MutFileDataType::XMLRoute1: {
584  xmltree document(_T("mutabor-routes"),true);
585  mutabor::RouteFactory::SaveRoutes(document);
586  document.Save(MurFileData.name.GetFullPath());
587 
588  }
589  break;
590  case MutFileDataType::Canceled:
591  case MutFileDataType::Unknown:
592  case MutFileDataType::LogicSource:
593  UNREACHABLE;
594  return;
595  }
596 }
