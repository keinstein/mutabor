422 void MutRouteWnd::CmRouteLoad(wxCommandEvent& event)
423 {
424  // it's our task
425  event.Skip(false);
426 
427  MutFileDataType filedata = FileNameDialog(wxGetApp().GetTopWindow(),
428  event.GetId());
429 
430 
431  switch (filedata.type) {
432  case MutFileDataType::Canceled:
433  return;
434  case MutFileDataType::TextRoute:
435  case MutFileDataType::UTF8TextRoute:
436  case MutFileDataType::XMLRoute1:
437  break;
438  case MutFileDataType::Unknown:
439  case MutFileDataType::LogicSource:
440  UNREACHABLE;
441  return;
442  }
443  if (!filedata.name.IsOk()) {
444  wxLogError(_("The string '%s' is not a valid file name."),
445  filedata.name.GetFullPath().c_str());
446  return;
447  }
448 
449  MurFileData = filedata;
450 
451 
452  switch (MurFileData.type) {
453  case MutFileDataType::TextRoute:
454  case MutFileDataType::UTF8TextRoute: {
455  wxFFile file(MurFileData.name.GetFullPath());
456  if (!file.IsOpened()) {
457  wxLogError(_("File '%s' couldn't be loaded"),MurFileData.name.GetFullPath().c_str());
458  return;
459  }
460  wxString text;
461  if (file.ReadAll(&text, wxConvUTF8)) {
462  MurFileData.type=MutFileDataType::UTF8TextRoute;
463  } else if (file.ReadAll(&text, wxConvISO8859_1)) {
464  MurFileData.type=MutFileDataType::TextRoute;
465  } else {
466  wxLogError(_("File '%s' could be opened, but not loaded."),
467  MurFileData.name.GetFullPath().c_str());
468  return;
469  }
470  compat30::LoadRoutes(text);
471  }
472  break;
473  case MutFileDataType::XMLRoute1: {
474  xmltree document("mutabor-routes",false);
475  if (!document.Load(MurFileData.name.GetFullPath())) {
476  wxLogError(_("File '%s' couldn't be loaded"),MurFileData.name.GetFullPath().c_str());
477  return;
478  }
479 
480  RouteClass::ClearRouteList();
481  InputDeviceClass::ClearDeviceList();
482  BoxClass::ClearBoxList();
483  OutputDeviceClass::ClearDeviceList();
484 
485  DebugCheckRoutes();
486 
487  mutabor::RouteFactory::LoadRoutes(document);
488  }
489  break;
490  case MutFileDataType::Canceled:
491  case MutFileDataType::Unknown:
492  case MutFileDataType::LogicSource:
493  UNREACHABLE;
494  return;
495  }
496 
497  // Activate the Route window
498  wxGetApp().CmRoutes(event);
499  MutRouteWnd * routewnd=NULL;
500  MutFrame * frame =
501  dynamic_cast<MutFrame*>(wxWindow::FindWindowById(WK_ROUTE));
502  if (frame) {
503  wxWindowList & list = frame->GetChildren();
504  for (wxWindowList::iterator i = list.begin();i!= list.end();i++) {
505  routewnd = dynamic_cast<MutRouteWnd *> (*i);
506  if (routewnd) break;
507  }
508  }
509  mutASSERT(routewnd);
510 
511  if (routewnd) {
512  routewnd->InitDevices();
513  routewnd->Layout();
514  routewnd->FitInside();
515  routewnd->Refresh();
516  }
517 
518  DebugCheckRoutes();
519 
520  wxCommandEvent ev(wxEVT_COMMAND_MENU_SELECTED,CM_ROUTES);
521  wxGetApp().ProcessEvent(ev);
522 }

524 void MutRouteWnd::CmRouteSave(wxCommandEvent& event)
525 {
526  TRACET(MutFrame);
527  event.Skip(false);
528 
529  switch (MurFileData.type) {
530  case MutFileDataType::Canceled:
531  return;
532  case MutFileDataType::Unknown:
533  CmRouteSaveAs(event);
534  return;
535  case MutFileDataType::TextRoute:
536  case MutFileDataType::UTF8TextRoute:
537  case MutFileDataType::XMLRoute1:
538  if (!MurFileData.name.IsOk() ) {
539  CmRouteSaveAs(event);
540  return;
541  }
542  break;
543  case MutFileDataType::LogicSource:
544  UNREACHABLE;
545  return;
546  }
547 
548  /* not applicable, since the file might not yet exsist

553  }
554  */
555 
556  switch (MurFileData.type) {
557  case MutFileDataType::TextRoute:
558  case MutFileDataType::UTF8TextRoute: {
559  wxFFile file(MurFileData.name.GetFullPath(), _T("w"));
560 
561  if (!file.IsOpened()) {
562  wxLogError(_("Cannot open routes file '%s' for writing."),

564  return;
565  }
566

582  MurFileData.name.GetFullPath().c_str());

584  }

587  xmltree document("mutabor-routes",true);

590 
591  }
