368  void MutSTCLexer::OnStyleNeeded(wxStyledTextEvent & event) {
369  int pos = event.GetPosition();
370  wxPoint editpos = editor->GetPosition();
371  int laststyled = editor->GetEndStyled();
372  DEBUGLOG(editlexer,
373  _T("event.pos = %d, editor pos = (%d,%d), last syled pos = %d"),
374  pos,editpos.x,editpos.y,laststyled);
375  int style = ERROR;
376  if (laststyled) {
377  style = editor->GetStyleAt(laststyled-1);
378  }
379 
380  int length = 0;
381  int lineNumber, level, oldlevel, levelstatus;
382  bool forceheader = false;
383 
384  lineNumber = editor->LineFromPosition(laststyled);
385  /* we may have an unfinished token or comment left over in the last run */
386  while (laststyled) {
387  int newstyle = editor->GetStyleAt(laststyled-1);
388  int newline = editor->LineFromPosition(laststyled-1);
389  if (newstyle != style && newline != lineNumber) break;
390  laststyled--;
391  style = newstyle;
392  lineNumber = newline;
393  }
394 
395  oldlevel = editor->GetFoldLevel(lineNumber) & wxSTC_FOLDLEVELNUMBERMASK;
396  DEBUGLOG(editlexer,_T("laststyed style = %x"),style);
397 
398  editor->StartStyling (laststyled, 0x1f);
399  wxCharBuffer text = editor->GetTextRangeRaw(laststyled,pos);
400 
401  // setting up folding data
402  if (laststyled) {
403  lineNumber = editor->LineFromPosition(laststyled-1);
404  level = editor->GetFoldLevel(lineNumber);
405  int newline = editor->LineFromPosition(laststyled);
406  levelstatus = level & ~wxSTC_FOLDLEVELNUMBERMASK;
407  level = level & wxSTC_FOLDLEVELNUMBERMASK;
408  if (newline != lineNumber) {
409  /* one character difference means '\n' */
410  levelstatus = (levelstatus & ~wxSTC_FOLDLEVELHEADERFLAG)
411  | wxSTC_FOLDLEVELWHITEFLAG;
412  lineNumber = newline;
413  }
414  } else {
415  lineNumber = editor->LineFromPosition(laststyled);
416  level = 0;
417  oldlevel = 0;
418  editor->SetFoldLevel(lineNumber,level);
419  levelstatus = wxSTC_FOLDLEVELWHITEFLAG;
420  }
421 
422  mutabor::mutabor_lexer lexer(text,pos-laststyled);
423  for (style = lexer.yylex(); style; style = lexer.yylex()) {
424 
425  bool isspace = false;
426 
427  switch (style) {
428  case MUTABOR_TOKEN_IDENTIFIER:
429  style = IDENTIFIER;
430  lexer.free_identifier();
431  break;
432 
433  case MUTABOR_TOKEN_F_NUMBER:
434  case MUTABOR_TOKEN_INTEGER:
435  style = NUMBER;
436  break;
437 
438  case MUTABOR_TOKEN_INTERVAL:
439  case MUTABOR_TOKEN_TONE:
440  case MUTABOR_TOKEN_TONESYSTEM:
441  case MUTABOR_TOKEN_RETUNING:
442  case MUTABOR_TOKEN_HARMONY:
443  case MUTABOR_TOKEN_LOGIC:
444  case MUTABOR_TOKEN_MIDICHANNEL:
445  levelstatus |= wxSTC_FOLDLEVELHEADERFLAG;
446  style = SECTIONKEYWORD;
447  forceheader = true;
448  level = 1;
449  oldlevel = 0;
450  break;
451 
452  case MUTABOR_TOKEN_ROOT:
453  case MUTABOR_TOKEN_MIDI_IN:
454  case MUTABOR_TOKEN_MIDI_OUT:
455  case '-':
456  case '+':
457  case '/':
458  case '*':
459  case '~':
460  case ':':
461  style = OPERATOR;
462  break;
463 
464  case MUTABOR_TOKEN_FORM:
465  case MUTABOR_TOKEN_KEY:
466  case MUTABOR_TOKEN_HARMONY_ANALYSIS:
467  style = RESERVEDWORD;
468  break;
469 
470  case MUTABOR_TOKEN_PARAMETER:
471  lexer.free_identifier();
472  case MUTABOR_TOKEN_DISTANCE:
473  case MUTABOR_TOKEN_ANCHOR:
474  case MUTABOR_TOKEN_ELSE:
475  case '@':
476  style = PARAMETER;
477  break;
478 
479  case MUTABOR_TOKEN_SPACES:
480  isspace = true;
481  style = DEFAULT;
482  break;
483 
484  case MUTABOR_TOKEN_OTHER:
485  case MUTABOR_TOKEN_CALLS:
486  case '=':
487  style = OTHER;
488  break;
489 
490  case ',':
491  style = DELIMITER;
492  break;
493 
494 
495  case MUTABOR_TOKEN_COMMENT:
496  style = COMMENT;
497  break;
498 
499  case MUTABOR_TOKEN_COMMENT_START:
500  if (level < oldlevel) forceheader = true;
501  levelstatus |= wxSTC_FOLDLEVELHEADERFLAG;
502  level++;
503  style = COMMENT;
504  break;
505 
506  case MUTABOR_TOKEN_COMMENT_END:
507  level--;
508  style = COMMENT;
509  break;
510 
511  case '[':
512  case '(':
513  case '<':
514  case '{':
515  if (level < oldlevel) forceheader = true;
516  levelstatus |= wxSTC_FOLDLEVELHEADERFLAG;
517  level++;
518  style = BRACE;
519  break;
520  case ']':
521  case ')':
522  case '>':
523  case '}':
524  style = BRACE;
525  level--;
526  break;
527 
528  case MUTABOR_TOKEN_ERROR:
529  case MUTABOR_TOKEN_ENDOFFILE: /* should not be reached, here */
530  default:
531  style = ERROR;
532  break;
533  }
534  length = lexer.YYLeng();
535  DEBUGLOG(editlexer,_T("flex style %d for %d chars: %s"),
536  style, length, ((wxString::FromUTF8(lexer.YYText(), length)).c_str()));
537 
538  laststyled = SetStyling(style, laststyled, length);
539  // updating lines
540  int endline = editor->LineFromPosition(laststyled);
541  if (!isspace)
542  levelstatus &= ~wxSTC_FOLDLEVELWHITEFLAG;
543  if (!forceheader && level <= oldlevel)
544  levelstatus &= ~wxSTC_FOLDLEVELHEADERFLAG;
545  editor->SetFoldLevel(lineNumber, levelstatus | oldlevel);
546  DEBUGLOG(editlexer,_T("line status: %x, level %d ‚Äì %d"),
547  editor->GetFoldLevel(lineNumber),
548  oldlevel,
549  editor->GetFoldLevel(lineNumber) & wxSTC_FOLDLEVELNUMBERMASK);
550

552  if (isspace)
553  tmplevelstatus |= wxSTC_FOLDLEVELWHITEFLAG;
554  if (lineNumber < endline)
555  oldlevel = level;
556  while (lineNumber < endline) {
557  forceheader = false;
558  levelstatus &= ~wxSTC_FOLDLEVELHEADERFLAG;
559  editor -> SetFoldLevel(++lineNumber, tmplevelstatus);
560  DEBUGLOG(editlexer,_T("line status: %x, level %d ‚Äì %d"),
561  editor->GetFoldLevel(lineNumber),
562  oldlevel,
563  editor->GetFoldLevel(lineNumber) & wxSTC_FOLDLEVELNUMBERMASK);
564  }
565  }
566 
567  event.Skip();
