398  void MutSTCLexer::OnStyleNeeded(wxStyledTextEvent & event) {
399  int pos = event.GetPosition();
400  int laststyled = editor->GetEndStyled();
401 #ifdef DEBUG
402  wxPoint editpos = editor->GetPosition();
403 #endif
404  DEBUGLOG(editlexer,
405  ("event.pos = %d, editor pos = (%d,%d), last syled pos = %d"),
406  pos,editpos.x,editpos.y,laststyled);
407  int style = ERROR;
408  if (laststyled) {
409  style = editor->GetStyleAt(laststyled-1);
410  }
411 
412  int length = 0;
413  int lineNumber, level, oldlevel, levelstatus;
414  bool forceheader = false;
415 
416  lineNumber = editor->LineFromPosition(laststyled);
417  /* we may have an unfinished token or comment left over in the last run */
418  while (laststyled) {
419  int newstyle = editor->GetStyleAt(laststyled-1);
420  int newline = editor->LineFromPosition(laststyled-1);
421  if (newstyle != style && newline != lineNumber) break;
422  laststyled--;
423  style = newstyle;
424  lineNumber = newline;
425  }
426 
427  oldlevel = editor->GetFoldLevel(lineNumber) & wxSTC_FOLDLEVELNUMBERMASK;
428  DEBUGLOG (editlexer, "laststyed style = %x" ,style);
429 
430  editor->StartStyling (laststyled, 0x1f);
431  wxCharBuffer text = editor->GetTextRangeRaw(laststyled,pos);
432 
433  // setting up folding data
434  if (laststyled) {
435  lineNumber = editor->LineFromPosition(laststyled-1);
436  level = editor->GetFoldLevel(lineNumber);
437  int newline = editor->LineFromPosition(laststyled);
438  levelstatus = level & ~wxSTC_FOLDLEVELNUMBERMASK;
439  level = level & wxSTC_FOLDLEVELNUMBERMASK;
440  if (newline != lineNumber) {
441  /* one character difference means '\n' */
442  levelstatus = (levelstatus & ~wxSTC_FOLDLEVELHEADERFLAG)
443  | wxSTC_FOLDLEVELWHITEFLAG;
444  lineNumber = newline;
445  }
446  } else {
447  lineNumber = editor->LineFromPosition(laststyled);
448  level = 0;
449  oldlevel = 0;
450  editor->SetFoldLevel(lineNumber,level);
451  levelstatus = wxSTC_FOLDLEVELWHITEFLAG;
452  }
453 
454  mutabor::mutabor_lexer lexer(text,pos-laststyled);
455  for (style = lexer.yylex(); style; style = lexer.yylex()) {
456 
457  bool isspace = false;
458 
459  switch (style) {
460  case MUTABOR_TOKEN_IDENTIFIER:
461  style = IDENTIFIER;
462  lexer.free_identifier();
463  break;
464 
465  case MUTABOR_TOKEN_F_NUMBER:
466  case MUTABOR_TOKEN_INTEGER:
467  style = NUMBER;
468  break;
469 
470  case MUTABOR_TOKEN_INTERVAL:
471  case MUTABOR_TOKEN_TONE:
472  case MUTABOR_TOKEN_TONESYSTEM:
473  case MUTABOR_TOKEN_RETUNING:
474  case MUTABOR_TOKEN_HARMONY:
475  case MUTABOR_TOKEN_LOGIC:
476  case MUTABOR_TOKEN_MIDICHANNEL:
477  levelstatus |= wxSTC_FOLDLEVELHEADERFLAG;
478  style = SECTIONKEYWORD;
479  forceheader = true;
480  level = 1;
481  oldlevel = 0;
482  break;
483 
484  case MUTABOR_TOKEN_ROOT:
485  case MUTABOR_TOKEN_MIDI_IN:
486  case MUTABOR_TOKEN_MIDI_OUT:
487  case '-':
488  case '+':
489  case '/':
490  case '*':
491  case '~':
492  case ':':
493  style = OPERATOR;
494  break;
495 
496  case MUTABOR_TOKEN_FORM:
497  case MUTABOR_TOKEN_KEY:
498  case MUTABOR_TOKEN_HARMONY_ANALYSIS:
499  style = RESERVEDWORD;
500  break;
501 
502  case MUTABOR_TOKEN_PARAMETER:
503  lexer.free_identifier();
504  FALLTHROUGH;
505  case MUTABOR_TOKEN_DISTANCE:
506  case MUTABOR_TOKEN_ANCHOR:
507  case MUTABOR_TOKEN_ELSE:
508  case '@':
509  style = PARAMETER;
510  break;
511 
512  case MUTABOR_TOKEN_SPACES:
513  isspace = true;
514  style = DEFAULT;
515  break;
516 
517  case MUTABOR_TOKEN_OTHER:
518  case MUTABOR_TOKEN_CALLS:
519  case '=':
520  style = OTHER;
521  break;
522 
523  case ',':
524  style = DELIMITER;
525  break;
526 
527 
528  case MUTABOR_TOKEN_COMMENT:
529  style = COMMENT;
530  break;
531 
532  case MUTABOR_TOKEN_COMMENT_START:
533  if (level < oldlevel) forceheader = true;
534  levelstatus |= wxSTC_FOLDLEVELHEADERFLAG;
535  level++;
536  style = COMMENT;
537  break;
538 
539  case MUTABOR_TOKEN_COMMENT_END:
540  level--;
541  style = COMMENT;
542  break;
543 
544  case '[':
545  case '(':
546  case '<':
547  case '{':
548  if (level < oldlevel) forceheader = true;
549  levelstatus |= wxSTC_FOLDLEVELHEADERFLAG;
550  level++;
551  style = BRACE;
552  break;
553  case ']':
554  case ')':
555  case '>':
556  case '}':
557  style = BRACE;
558  level--;
559  break;
560 
561  case MUTABOR_TOKEN_ERROR:
562  case MUTABOR_TOKEN_ENDOFFILE: /* should not be reached, here */
563  default:
564  style = ERROR;
565  break;
566  }
567  length = lexer.YYLeng();
568  DEBUGLOG (editlexer, "flex style %d for %d chars: %s" ,
569  style, length, ((wxString::FromUTF8(lexer.YYText(), length)).c_str()));
570 
571  laststyled = SetStyling(style, laststyled, length);
572  // updating lines
573  int endline = editor->LineFromPosition(laststyled);
574  if (!isspace)
575  levelstatus &= ~wxSTC_FOLDLEVELWHITEFLAG;
576  if (!forceheader && level <= oldlevel)
577  levelstatus &= ~wxSTC_FOLDLEVELHEADERFLAG;
578  editor->SetFoldLevel(lineNumber, levelstatus | oldlevel);
579  DEBUGLOG (editlexer, "line status: %x, level %d – %d" ,
580  editor->GetFoldLevel(lineNumber),
581  oldlevel,
582  editor->GetFoldLevel(lineNumber) & wxSTC_FOLDLEVELNUMBERMASK);
583 
584  int tmplevelstatus = (levelstatus & ~wxSTC_FOLDLEVELHEADERFLAG) | level;
585  if (isspace)
586  tmplevelstatus |= wxSTC_FOLDLEVELWHITEFLAG;
587  if (lineNumber < endline)
588  oldlevel = level;
589  while (lineNumber < endline) {
590  forceheader = false;
591  levelstatus &= ~wxSTC_FOLDLEVELHEADERFLAG;
592  editor -> SetFoldLevel(++lineNumber, tmplevelstatus);
593  DEBUGLOG (editlexer, "line status: %x, level %d – %d" ,
594  editor->GetFoldLevel(lineNumber),
595  oldlevel,
596  editor->GetFoldLevel(lineNumber) & wxSTC_FOLDLEVELNUMBERMASK);
597  }
598  }
599 
600  event.Skip();
601  }

617  if ((editor->GetStyleAt(i) & 0x1f) != style) {
618  DEBUGLOG(editlexer,
619  ("checking style at %d (%d – %d), %x == %x"),
620  (int)i,(int)pos,(int)pos+length,(int)editor->GetStyleAt(i),(int)style);
