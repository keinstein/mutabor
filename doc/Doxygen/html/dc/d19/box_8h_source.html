181 inline mutabor_note_type * mutabor_find_key_in_box(mutabor_box_type * box, size_t index) {
182  mutabor_key_index_type *plane_ptr;
183  size_t plane;
184  mutabor_check_key_count(box);
185  if (!box->key_count || index == MUTABOR_NO_NEXT) return NULL;
186  plane = index >> MUT_BOX_MAX_KEY_INDEX_LOG;
187  plane_ptr = &(box->current_keys);
188  while (plane-- && plane_ptr) plane_ptr = plane_ptr->next;
189  if (!plane_ptr) {
190  return NULL;
191  }
192  return &(plane_ptr->key[index & MUT_BOX_MAX_KEY_INDEX]);
193 }

223 inline void mutabor_delete_key_in_box(mutabor_box_type * box, size_t index) {
224  mutabor_note_type *last_key, *current_key;
225  size_t last_index;
226 
227  if (!box->key_count) return;
228 
229  current_key = mutabor_find_key_in_box(box,index);
230  if (current_key == NULL) {
231  UNREACHABLE;
232  return;
233  }
234 
235  if (!index) {
236 
237  if (current_key -> next != MUTABOR_NO_NEXT) {
238  mutabor_note_type * next_key
239  = mutabor_find_key_in_box(box,current_key->next);
240  if (current_key->next == box->last_key)
241  box -> last_key = 0;
242  *(current_key) = *(next_key);
243  next_key->next = MUTABOR_NO_NEXT;
244  next_key->userdata = NULL;
245  box->key_count--;
246 
247  } else {
248  mutASSERT(box->key_count == 1);
249  current_key->userdata = NULL;
250  box->key_count = 0;
251  }
252  return;
253  }
254 
255 
256 
257  DEBUGLOG2(kernel_box,("index = %d"),(int)index);
258  last_key = mutabor_find_key_in_box(box,0);
259  last_index = 0;
260  while (last_key != NULL && last_key -> next != index) {
261 /* DEBUGLOG2(kernel_box,_T("last_key->next = %d"), last_key->next); */
262  last_index = last_key->next;
263  last_key = mutabor_find_key_in_box(box,last_index);
264  }
265  if (last_key == NULL || last_key -> next != index) {
266  mutabor_check_key_count(box);
267  return;
268  }
269  if (index == box->last_key) {
270  mutASSERT(last_index != box->last_key);
271  box->last_key = last_index;
272  }
273 
274  last_key->next = current_key->next;
275  current_key -> next = MUTABOR_NO_NEXT;
276  current_key -> userdata = NULL;
277  box->key_count--;
278  mutabor_check_key_count(box);
279 }

282 inline mutabor_note_type * mutabor_create_key_in_box (mutabor_box_type * box) {
283  mutabor_note_type * last, *new_key = NULL;
284  mutabor_key_index_type *plane;
285  mutabor_key_index_type *oldplane;
286  size_t index = 0;
287  size_t planeidx = 0;
288  bool found = false;
289 
290  mutASSERT(box);
291 
292  if (!box->key_count) {
293  box->key_count = 1;
294  mutASSERT(box->current_keys.key[0].userdata == NULL);
295  mutASSERT(box->current_keys.key[0].next == MUTABOR_NO_NEXT);
296  return &(box->current_keys.key[0]);
297  }
298 
299  mutabor_check_key_count(box);
300 
301  last = mutabor_find_key_in_box(box,box->last_key);
302  if (last == NULL) {
303  UNREACHABLE;
304  return NULL;
305  }
306  oldplane = plane = &(box->current_keys);
307 
308  while (plane != NULL && new_key == NULL) {
309  size_t i;
310  size_t curridx = planeidx << MUT_BOX_MAX_KEY_INDEX_LOG;
311  for (i = 0 ; i <= MUT_BOX_MAX_KEY_INDEX; i++) {
312  if (i+curridx == box->last_key) {
313  continue;
314  }
315  if (plane->key[i].next == MUTABOR_NO_NEXT) {
316  index = i + curridx;
317  new_key = &(plane->key[i]);
318  found = true;
319  break;
320  }
321  }
322  if (found) break;
323  planeidx++;
324  oldplane = plane;
325  plane = plane->next;
326  }
327  if (new_key == NULL) {
328  plane = oldplane->next = (mutabor_key_index_type *)malloc(sizeof(mutabor_key_index_type));
329  if (plane == NULL) {
330  mutabor_check_key_count(box);
331  return NULL;
332  }
333  mutabor_initialize_keyplane(plane);
334  new_key = &(plane->key[0]);
335  index = planeidx << MUT_BOX_MAX_KEY_INDEX_LOG;
336  }
337 
338  box -> last_key = last -> next = index;
339  new_key -> next = MUTABOR_NO_NEXT;
340  box->key_count++;
341  mutabor_check_key_count(box);
342  return new_key;
343 }
