208 inline mutabor_note_type * mutabor_find_key_in_box(mutabor_box_type * box, size_t index) {
209  mutabor_key_index_type *plane_ptr;
210  size_t plane;
211  mutabor_check_key_count(box);
212  if (!box->key_count || index == MUTABOR_NO_NEXT) return NULL;
213  plane = index >> MUT_BOX_MAX_KEY_INDEX_LOG;
214  plane_ptr = &(box->current_keys);
215  while (plane-- && plane_ptr) plane_ptr = plane_ptr->next;
216  if (!plane_ptr) {
217  return NULL;
218  }
219  return &(plane_ptr->key[index & MUT_BOX_MAX_KEY_INDEX]);
220 }

250 static inline void mutabor_delete_key_in_box(mutabor_box_type * box, size_t index) {
251  mutabor_note_type *last_key, *current_key;
252  size_t last_index;
253 
254  if (!box->key_count) return;
255 
256  current_key = mutabor_find_key_in_box(box,index);
257  if (current_key == NULL) {
258  UNREACHABLE;
259  return;
260  }
261 
262  if (!index) {
263 
264  if (current_key -> next != MUTABOR_NO_NEXT) {
265  mutabor_note_type * next_key
266  = mutabor_find_key_in_box(box,current_key->next);
267  if (current_key->next == box->last_key)
268  box -> last_key = 0;
269  *(current_key) = *(next_key);
270  next_key->next = MUTABOR_NO_NEXT;
271  next_key->userdata = NULL;
272  box->key_count--;
273 
274  } else {
275  mutASSERT(box->key_count == 1);
276  current_key->userdata = NULL;
277  box->key_count = 0;
278  }
279  return;
280  }

282 
283 
284  DEBUGLOG2(kernel_box,("index = %d"),(int)index);
285  last_key = mutabor_find_key_in_box(box,0);
286  last_index = 0;
287  while (last_key != NULL && last_key -> next != index) {
288 /* DEBUGLOG2(kernel_box,_T("last_key->next = %d"), last_key->next); */
289  last_index = last_key->next;
290  last_key = mutabor_find_key_in_box(box,last_index);
291  }
292  if (last_key == NULL || last_key -> next != index) {
293  mutabor_check_key_count(box);
294  return;
295  }
296  if (index == box->last_key) {
297  mutASSERT(last_index != box->last_key);
298  box->last_key = last_index;
299  }
300

302  current_key -> next = MUTABOR_NO_NEXT;
303  current_key -> userdata = NULL;
304  box->key_count--;
305  mutabor_check_key_count(box);
306 }

309 static inline mutabor_note_type * mutabor_create_key_in_box (mutabor_box_type * box) {
310  mutabor_note_type * last, *new_key = NULL;
311  mutabor_key_index_type *plane;
312  mutabor_key_index_type *oldplane;
313  size_t index = 0;
314  size_t planeidx = 0;
315  bool found = false;
316 
317  mutASSERT(box);
318 
319  if (!box->key_count) {
320  box->key_count = 1;
321  mutASSERT(box->current_keys.key[0].userdata == NULL);
322  mutASSERT(box->current_keys.key[0].next == MUTABOR_NO_NEXT);
323  return &(box->current_keys.key[0]);
324  }
325 
326  mutabor_check_key_count(box);
327 
328  last = mutabor_find_key_in_box(box,box->last_key);
329  if (last == NULL) {
330  UNREACHABLE;
331  return NULL;
332  }
333  oldplane = plane = &(box->current_keys);
334 
335  while (plane != NULL && new_key == NULL) {
336  size_t i;
337  size_t curridx = planeidx << MUT_BOX_MAX_KEY_INDEX_LOG;
338  for (i = 0 ; i <= MUT_BOX_MAX_KEY_INDEX; i++) {
339  if (i+curridx == box->last_key) {
340  continue;
341  }
342  if (plane->key[i].next == MUTABOR_NO_NEXT) {
343  index = i + curridx;
344  new_key = &(plane->key[i]);
345  found = true;
346  break;
347  }
348  }
349  if (found) break;
350  planeidx++;
351  oldplane = plane;
352  plane = plane->next;
353  }
354  if (new_key == NULL) {
355  plane = oldplane->next = (mutabor_key_index_type *)malloc(sizeof(mutabor_key_index_type));
356  if (plane == NULL) {
357  mutabor_check_key_count(box);
358  return NULL;
359  }
360  mutabor_initialize_keyplane(plane);
361  new_key = &(plane->key[0]);
362  index = planeidx << MUT_BOX_MAX_KEY_INDEX_LOG;
363  }
364 
365  box -> last_key = last -> next = index;
366  new_key -> next = MUTABOR_NO_NEXT;
367  box->key_count++;
368  mutabor_check_key_count(box);
369  return new_key;
370 }
