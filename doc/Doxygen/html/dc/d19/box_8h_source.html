215 inline mutabor_note_type * mutabor_find_key_in_box(mutabor_box_type * box, size_t index) {
216  mutabor_key_index_type *plane_ptr;
217  size_t plane;
218  mutabor_check_key_count(box);
219  if (!box->key_count || index == MUTABOR_NO_NEXT) return NULL;
220  plane = index >> MUT_BOX_MAX_KEY_INDEX_LOG;
221  plane_ptr = &(box->current_keys);
222  while (plane-- && plane_ptr) plane_ptr = plane_ptr->next;
223  if (!plane_ptr) {
224  return NULL;
225  }
226  return &(plane_ptr->key[index & MUT_BOX_MAX_KEY_INDEX]);
227 }

258  mutabor_note_type *last_key, *current_key;
259 
260  if (!box->key_count) return;
261 
262  current_key = mutabor_find_key_in_box(box,index);
263  if (current_key == NULL) {
264  UNREACHABLE;
265  return;
266  }
267 
268  if (!index) {
269 
270  if (current_key -> next != MUTABOR_NO_NEXT) {
271  mutabor_note_type * next_key
272  = mutabor_find_key_in_box(box,current_key->next);
273  if (current_key->next == box->last_key)
274  box -> last_key = 0;
275  *(current_key) = *(next_key);
276  next_key->next = MUTABOR_NO_NEXT;
277  next_key->userdata = NULL;
278  box->key_count--;
279 
280  } else {

315 inline mutabor_note_type * mutabor_create_key_in_box (mutabor_box_type * box) {
316  mutabor_note_type * last, *new_key = NULL;
317  mutabor_key_index_type *plane;
318  mutabor_key_index_type *oldplane;
319  size_t index = 0;
320  size_t planeidx = 0;
321  bool found = false;
322 
323  mutASSERT(box);
324 
325  if (!box->key_count) {
326  box->key_count = 1;
327  mutASSERT(box->current_keys.key[0].userdata == NULL);
328  mutASSERT(box->current_keys.key[0].next == MUTABOR_NO_NEXT);
329  return &(box->current_keys.key[0]);
330  }
331 
332  mutabor_check_key_count(box);
333 
334  last = mutabor_find_key_in_box(box,box->last_key);
335  if (last == NULL) {
336  UNREACHABLE;
337  return NULL;
338  }
339  oldplane = plane = &(box->current_keys);
340 
341  while (plane != NULL && new_key == NULL) {
342  size_t i;
343  size_t curridx = planeidx << MUT_BOX_MAX_KEY_INDEX_LOG;
344  for (i = 0 ; i <= MUT_BOX_MAX_KEY_INDEX; i++) {
345  if (i+curridx == box->last_key) {
346  continue;
347  }
348  if (plane->key[i].next == MUTABOR_NO_NEXT) {
349  index = i + curridx;
350  new_key = &(plane->key[i]);
351  found = true;
352  break;
353  }
354  }
355  if (found) break;
356  planeidx++;
357  oldplane = plane;
358  plane = plane->next;
359  }
360  if (new_key == NULL) {
361  plane = oldplane->next = (mutabor_key_index_type *)malloc(sizeof(mutabor_key_index_type));
362  if (plane == NULL) {
363  mutabor_check_key_count(box);
364  return NULL;
365  }
366  mutabor_initialize_keyplane(plane);
367  new_key = &(plane->key[0]);
368  index = planeidx << MUT_BOX_MAX_KEY_INDEX_LOG;
369  }
370 
371  box -> last_key = last -> next = index;
372  new_key -> next = MUTABOR_NO_NEXT;
373  box->key_count++;
374  mutabor_check_key_count(box);
375  return new_key;
376 }
