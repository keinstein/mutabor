331  inline int get_data_size(uint8_t status_byte) {
332  static const int midi_size[] = { 3, 3, 3, 3, 2, 2, 3,-1 };
333  static const int system_size[] = {-1, 2, 3, 2, 1, 1, 1, -1,
334  1, 1, 1, 1, 1, 1, 1, -1 };
335 
336  if (!(status_byte & 0x80)) return -1;
337  int size = midi_size [(status_byte >> 4) & 0x7];
338  return size == -1 ? system_size[status_byte & 0x0F]:size;
339  }

412  inline int get_default_controller_value(int ctrl) {
413  /* The corresponding specification is RP
414  15. It is availlable at
415  http://www.midi.org/techspecs/rp15.php */
416  if ((ctrl >= midi::FIRST_RPN && ctrl <= midi::LAST_RPN)
417  || (ctrl >= midi::FIRST_NRPN && ctrl <= midi::LAST_NRPN)) {
418  return CONTROLLER_UNKNOWN;
419  }
420  switch (ctrl) {
421  case EXPRESSION_COARSE:
422  case EXPRESSION_FINE:
423  case NON_REGISTERED_PARAMETER_FINE:
424  case NON_REGISTERED_PARAMETER_COARSE:
425  case REGISTERED_PARAMETER_FINE:
426  case REGISTERED_PARAMETER_COARSE:
427  return CONTROLLER_MAX;
428  case BALANCE_COARSE:
429  return CONTROLLER_CENTRE;
430  case BANK_FINE:
431  case BANK_COARSE:
432  case VOLUME_COARSE:
433  case PAN_POSITION_COARSE:
434  case EFFECTS_LEVEL:
435  case TREMULO_LEVEL:
436  case CHORUS_LEVEL:
437  case CELESTE_LEVEL:
438  case PHASER_LEVEL:
439  case SOUND_VARIATION:
440  case SOUND_TIMBRE:
441  case SOUND_RELEASE_TIME:
442  case SOUND_ATTACK_TIME:
443  case SOUND_BRIGHTNESS:
444  case SOUND_DECAY_TIME:
445  case SOUND_VIBRATO_RATE:
446  case SOUND_VIBRATO_DEPTH:
447  case SOUND_VIBRATO_DELAY:
448  case SOUND_10:
449  return CONTROLLER_UNKNOWN;
450  case DATA_ENTRY_COARSE:
451  case DATA_ENTRY_FINE:
452  case DATA_BUTTON_INCREMENT:
453  case DATA_BUTTON_DECREMENT:
454  case ALL_SOUND_OFF:
455  case ALL_CONTROLLERS_OFF:
456  case LOCAL_ON_OFF:
457  case ALL_NOTES_OFF:
458  case OMNI_OFF:
459  case OMNI_ON:
460  case MONO_ON:
461  case POLY_ON:
462  // do not reset
463  return CONTROLLER_UNKNOWN_NOSEND;
464  case MODULATION_WHEEL_COARSE:
465  case MODULATION_WHEEL_FINE:
466  default:
467  break;
468  }
469  return 0;
