249  if ( ('A' < event.GetKeyCode() && event.GetKeyCode() <= 'z')
250  || event.GetKeyCode() == WXK_TAB
251  || event.GetKeyCode() == WXK_RETURN
252  || event.GetKeyCode() == WXK_SPACE ){
253  if ((box) && ToGUIBase(box)) {
254  MutLogicWnd * logic = ToGUIBase(box)->GetLogicWindow();
255  if (logic)
256  wxPostEvent(logic->GetEventHandler(),event);
257  else {
258  UNREACHABLEC;

260  }
261  }
262  else event.Skip();
263 
264  SetCurrentBox(box);
265  mutASSERT(box);
