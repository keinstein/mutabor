224  void WriteNumber(size_t count) {
225  uint8_t tmp[4];
226  int i = 0;
227  tmp[0] = count & 0x7f;
228  while (count && i < 4) {
229  tmp[i++] = count & 0x7f;

231  }

304  template <class i>
305  void SendSysExCont (i from, i to,
306  mutint64 delta = MUTABOR_NO_DELTA,
307  size_t offset = 0)
308  {
309 
310 
311  if (from == to) {
312  boost::throw_exception(message_length_error(
313  gettext_noop("The system exclusive message must contain at least a device id.")));
314  }
315 
316  int tmp_sysex_id = *from ;
317  if (tmp_sysex_id == 0) {
318  i myfrom = from + 1;
319  if (to - myfrom < 2) {
320  boost::throw_exception(wrong_id(gettext_noop("System exclusive message contains an invalid device id")));
321  }
322  tmp_sysex_id = 0x10000 | ((*(myfrom++)) << 8);
323  tmp_sysex_id |= ((*myfrom) << 8);
324  }
325 
326  if (!running_sysex) {
327  sysex_id = tmp_sysex_id;
328  } else if (sysex_id != tmp_sysex_id) {
329  boost::throw_exception(wrong_id(gettext_noop("Device id of continuation package deos not match the id of the system exclusive message")));
330  }
331 
332  if (delta == MUTABOR_NO_DELTA) {
333  WriteDelta();
334  } else {
335  WriteNumber(delta);
336  }
337 
338  push_back(running_sysex?midi::SYSEX_END:midi::SYSEX_START);
339  WriteLongChunk(from, to, offset);
340 
341  if (offset == 0 && *(to - 1) != midi::SYSEX_END) {
342  running_sysex = true;
343  } else {
344  running_sysex = false;
345  }
346  }
