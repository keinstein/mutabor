121  template <typename T>
122  void WriteNumber(T count) {
123  if (!midi::is_valid_length(count))
124  BOOST_THROW_EXCEPTION(delta_length_error(_mutN("trying to write number > 0x0FFFFFFF")));
125  uint8_t tmp[4];
126  int i = 1;
127  tmp[0] = count & 0x7f;
128  count >>= 7;
129  while (count && i < 4) {
130  tmp[i++] = (count & 0x7f ) | 0x80;
131  count >>= 7;
132  }
133  check_capacity(i);
134  while (i)
135  push_back(tmp[--i]);
136  }

535  template <class i>
536  void SendSysExCont (i from, i to,
537  microseconds delta = InputDeviceClass::NO_DELTA(),
538  size_t offset = 0)
539  {
540 
541 
542  if (from == to) {
543  BOOST_THROW_EXCEPTION(message_length_error(
544  _mutN("The system exclusive message must contain at least a device id.")));
545  }
546 
547  int tmp_sysex_id = *from ;
548  if (tmp_sysex_id == 0) {
549  i myfrom = from + 1;
550  if (to - myfrom < 2) {
551  BOOST_THROW_EXCEPTION(wrong_id(_mutN("System exclusive message contains an invalid device id")));
552  }
553  tmp_sysex_id = 0x10000 | ((*(myfrom++)) << 8);
554  tmp_sysex_id |= ((*myfrom) << 8);
555  }
556 
557  if (!running_sysex) {
558  sysex_id = tmp_sysex_id;
559  } else if (sysex_id != tmp_sysex_id) {
560  BOOST_THROW_EXCEPTION(wrong_id(_mutN("Device id of continuation package deos not match the id of the system exclusive message")));
561  }
562 
563 
564  WriteDelta(delta);
565 
566  base::SendSysExCont(running_sysex, from, to, offset);
567 
568  if (offset == 0 && *(to - 1) != midi::SYSEX_END) {
569  running_sysex = true;
570  } else {
571  running_sysex = false;
572  }
573  }

765  void Save(std::ostream &os) {
766  std::pair<uint8_t,uint8_t > ticks = Tracks.begin()->second->getMidiIickSignature();
767  size_t number = Tracks.size();
768 
769  // Let's try to generate Format 0 files: 1 Track including all information.
770 
771  uint8_t Header[14] = {
772  // File header
773  'M', 'T', 'h', 'd', // Chunk type MIDI file header
774  0, 0, 0, 6, // chunk length
775  0, 2, // file type Format 2 = independent tracks
776  uint8_t((number & 0xFF00) >> 8) , uint8_t(number & 0xFF), // number of tracks
777  ticks.first, ticks.second // Ticks per qaurter or second
778  /* Alternatve format last 2 bytes:
779  0x80 & negative_SMPTE_format_in_7bit,
780  ticks_per_frame */
781  };
782 
783  os.write(reinterpret_cast<char *>(Header), 14);
784  for (auto &i : Tracks)
785  i.second->Save(os);
