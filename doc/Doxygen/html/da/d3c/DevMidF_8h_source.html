224  void WriteNumber(size_t count) {
225  uint8_t tmp[4];
226  int i = 0;
227  tmp[0] = count & 0x7f;
228  while (count && i < 4) {
229  tmp[i++] = count & 0x7f;
230  count >>= 7;
231  }
232  if (count)
233  throw delta_length_error(gettext_noop("trying to write number > 0x0FFFFFFF"));
234

236  while (--i)
237  push_back(tmp[i] | 0x80);
238  push_back(tmp[0]);

306  template <class i>
307  void SendSysExCont (i from, i to,
308  mutint64 delta = MUTABOR_NO_DELTA,
309  size_t offset = 0)
310  {
311 
312 
313  if (from == to) {
314  throw message_length_error(
315  gettext_noop("The system exclusive message must contain at least a device id."));
316  }
317 
318  int tmp_sysex_id = *from ;
319  if (tmp_sysex_id == 0) {
320  i myfrom = from + 1;
321  if (to - myfrom < 2) {
322  throw wrong_id(gettext_noop("System exclusive message contains an invalid device id"));
323  }
324  tmp_sysex_id = 0x10000 | ((*(myfrom++)) << 8);
325  tmp_sysex_id |= ((*myfrom) << 8);
326  }
327 
328  if (!running_sysex) {
329  sysex_id = tmp_sysex_id;
330  } else if (sysex_id != tmp_sysex_id) {
331  throw wrong_id(gettext_noop("Device id of continuation package deos not match the id of the system exclusive message"));
332  }
333 
334  if (delta == MUTABOR_NO_DELTA) {
335  WriteDelta();
336  } else {
337  WriteNumber(delta);
338  }
339 
340  push_back(running_sysex?midi::SYSEX_END:midi::SYSEX_START);
341  WriteLongChunk(from, to, offset);
342 
343  if (offset == 0 && *(to - 1) != midi::SYSEX_END) {
344  running_sysex = true;
345  } else {
346  running_sysex = false;
347  }
348  }
