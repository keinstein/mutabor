149  inline std::string DEMANGLE(const std::string & x) {
150  int status;
151  char * dem = abi::__cxa_demangle(x.c_str(), 0, 0, &status);
152  std::string retval;
153  switch (status) {
154  case 0: /* success */
155  if (dem) {
156  retval = dem;
157  free(dem);
158  }
159  break;
160  case -1: /* Memory allocation failiure */
161  case -2: /* not a valid C++ name */
162  case -3: /* invalid argument */
163  retval = x;
164  break;
165  }
166  return retval;
167  }
