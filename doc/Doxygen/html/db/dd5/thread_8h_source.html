510  class meeting_point {
511  public:
512  enum {
513  locked_master=1,
514  locked_slave=2,
515  unlocked_first=4,
516  unlocked_secound=8,
517  locked_mask = locked_master | locked_slave,
518  unlocked_mask = unlocked_first | unlocked_secound
519  };
520  meeting_point():sync_condition() {
521  locked = 0;
522  }
523 
524  void sync(int role,
525  boost::unique_lock<boost::mutex> & lock) {
526  // We must ensure that subsequent syncronisation
527  // points do not interfere with each other.
528  while (locked & unlocked_mask) {
529  sync_condition.wait(lock);
530  }
531  locked |= role;
532  if ((locked & locked_mask) == locked_mask) {
533  locked |= (unlocked_mask & ~(int) unlocked_secound);
534  } else {
535  while (!(locked & unlocked_mask)) {
536  sync_condition.wait(lock);
537  }
538  locked &= ~(int) unlocked_first;
539  }
540  locked &= ~role;
541  sync_condition.notify_all();
542  }
543 
544  protected:
545  boost::condition_variable sync_condition;
546  boost::atomic<int> locked;
547  };
