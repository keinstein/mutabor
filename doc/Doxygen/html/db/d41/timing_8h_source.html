254  static mutint64 update_duration_midi(mutint64 duration,
255  const timing_params & oldd,
256  const timing_params & newd) {
257  if ((oldd.is_fixed_ticks && newd.is_fixed_ticks) || duration == MUTABOR_NO_DELTA) return duration;
258  mutint64 old_quarter = oldd.is_fixed_ticks ? 1000000 : oldd.quarter_duration;
259  mutint64 new_quarter = newd.is_fixed_ticks ? 1000000 : newd.quarter_duration;
260  mutint64 old_factor = old_quarter/oldd.quarter_divisions;
261  mutint64 new_factor = new_quarter/newd.quarter_divisions;
262 
263  return ((duration + (old_factor+1)/2) / old_factor) * new_factor;
264  }

288  static mutint64 update_duration_exact(mutint64 duration,
289  const timing_params & oldd,
290  const timing_params & newd) {
291  if ((oldd.is_fixed_ticks && newd.is_fixed_ticks) || duration == MUTABOR_NO_DELTA) return duration;
292  mutint64 old_quarter = oldd.is_fixed_ticks ? 1000000 : oldd.quarter_duration;
293  mutint64 new_quarter = newd.is_fixed_ticks ? 1000000 : newd.quarter_duration;
294  mutint64 delta_quarter = new_quarter - old_quarter;
295  if (delta_quarter > 0) {
296  duration = (duration * new_quarter)/old_quarter;
297  }

299  duration = (duration * oldd.quarter_divisions)/newd.quarter_divisions;

302  duration = (duration * new_quarter)/old_quarter;
303  }

364  void Start(mutint64 t = 0) {
365  clock_getres(CLOCK_MONOTONIC,&resolution);
366  clock_gettime(CLOCK_MONOTONIC,&start);
367  start.tv_sec -= t/(1000*1000);
368  start.tv_nsec -= (t % (1000*1000))*1000;
369  if (start.tv_nsec < 0) {
370  start.tv_sec -= 1;
371  start.tv_nsec += 1000*1000*1000;
372  } else if (start.tv_nsec > 1000*1000*1000) {
373  start.tv_nsec -= 1000*1000*1000;
374  start.tv_sec += 1;
375  }
376  DEBUGLOGTYPE(timer,
377  CurrentTimerBase,

380  start.tv_nsec);
