98  void SetRouteType(mutabor::RouteType routetype) {
99  wxPanel * panel = NULL;
100  switch (routetype) {
101  case mutabor::RTall: panel = all_panel; break;
102  case mutabor::RTchannel: panel = channel_panel; break;
103  case mutabor::RTstaff: panel = track_panel; break;
104  case mutabor::RTelse: panel = else_panel; break;
105  default:
106  std::cerr << "MidiFileInputFilterPanel::SetRouteType: invaid route type "
107  << routetype << std::endl;
108  abort();
109  }
110  for (size_t i = 0 ; i < type->GetPageCount(); i++) {
111  if (type -> GetPage(i) != panel) continue;
112  type->SetSelection(i);
113  }
114  }
