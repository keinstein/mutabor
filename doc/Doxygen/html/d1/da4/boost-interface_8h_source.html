55 template<bool locking=1>
56 class intrusive_ptr_refcount_type {
57 public:
58  typedef mutabor::ScopedCondLock<locking> locker;
59  intrusive_ptr_refcount_type():value(0) {}
60 
61  void lock () {
62  if (locking)
63  mutex.Lock();
64  }
65 
66  void unlock () {
67  if (locking)
68  mutex.Unlock();
69  }

71  size_t operator ++() {
72  locker lock(mutex);
73  return ++value;
74  }

78  return value++;
79  }

88  return value--;

92  locker lock(const_cast<intrusive_ptr_refcount_type *>(this)->mutex);
