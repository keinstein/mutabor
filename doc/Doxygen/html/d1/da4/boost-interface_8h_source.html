155 {
156  if (!obj) return;
157  obj->intrusive_ptr_refcount.fetch_add(1,boost::memory_order_relaxed);
158  DEBUGLOGTYPE(smartptr,*obj,("Incrementing %p to %d"),

163 template <class intrusive_ptr_T>
164 inline void intrusive_ptr_release(intrusive_ptr_T * obj)
165 {
166  if (!obj) return;
167  DEBUGLOGTYPE(smartptr,*obj,("Decrementing %p from %d"),
168  (void *)obj,(int)intrusive_ptr_get_refcount(obj));
169  // print_stacktrace(isDebugFlag(smartptr));
170  if (obj->intrusive_ptr_refcount.fetch_sub(1, boost::memory_order_release) == 1) {
171  intrusive_ptr_atomic_fence(obj);
172  delete obj;
173  }
