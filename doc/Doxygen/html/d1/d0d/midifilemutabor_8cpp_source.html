173 int main(int ac, char* av[])
174 {
175  try {
176  mutabor::InitDeviceFactories();
177 
178  boost::locale::generator gen;
179  gen.add_messages_path(LOCALEDIR);
180  gen.add_messages_domain("mutabor");
181  locale::global(gen(""));
182  po::options_description desc(_mut("Allowed options"));
183  po::positional_options_description p;
184  po::variables_map vm;
185 
186  add_options(desc,p,vm);
187 
188 
189  try {
190  po::store(po::command_line_parser(ac, av).
191  options(desc).positional(p).run(), vm);
192  po::notify(vm);
193  }
194  catch (const po::too_many_positional_options_error & e) {
195  cerr << STR_too_many_positional_options_error
196  << std::endl
197  << std::endl;
198  do_help(desc,*av);
199  return 1;
200  }
201  mut_catch_exception(multiple_values)
202  mut_catch_exception(multiple_occurrences)
203  mut_catch_exception(required_option)
204  mut_catch_exception(unknown_option)
205  mut_catch_exception(ambiguous_option)
206  catch(const exception& e) {
207  cerr << "error: " << e.what() << "\n";
208  do_help(desc,*av);
209  return 1;
210  }
211 
212  route = mutabor::RouteFactory::Create();
213  box = mutabor::BoxFactory::Create(mutabor::Box0,0);
214  connect(route, box);
215 
216  mutabor::CurrentTime.UseRealtime(false);
217  outguard =mutabor::DeviceFactory::CreateOutput(mutabor::DTMidiFile);
218  out = static_cast<mutabor::OutputMidiFile *>(outguard.get());
219  out->SetBendingRange(2);
220  //out = new midicmnOutputDevice(3,("Test"));
221 
222  connect(route, outguard);
223  route->SetOutputFrom (0);
224  route->SetOutputTo (15);
225  route->OutputAvoidDrumChannel (true);
226 
227 
228 
229  inguard = (mutabor::DeviceFactory::CreateInput(mutabor::DTMidiFile));
230  in = static_cast<mutabor::InputMidiFile *> (inguard.get());
231 
232  connect(route, inguard);
233 
234 
235  if (vm.count("help")) {
236  do_help(desc,*av);
237  return 0;
238  }
239 
240  if (inputfile.empty() || outputfile.empty()) {
241  do_help(desc,*av);
242  return 1;
243  }
244  in -> SetName(inputfile);
245  out -> SetName(outputfile);
246 
247  errno = 0;
248  std::ifstream file(logicfile.c_str());
249  if (!file.is_open()) {
250  cerr << boost::format(_mut("Error: Could not open logic file `%s'.")) % logicfile
251  << std::endl;
252  cerr << strerror(errno) << std::endl;
253  return 1;
254  }
255  file.seekg(0, std::ios::end);
256  std::streampos length = file.tellg();
257  file.seekg(0, std::ios::beg);
258  char * logic_string = (char *) malloc((size_t)length+1);
259  if (!(file.read(logic_string, length).gcount() == length)) {
260  cerr << boost::format(_mut("Error: Could not read logic file `%s'.")) % logicfile
261  << std::endl;
262  cerr << strerror(errno) << std::endl;
263  return 1;
264  }
265  logic_string[length] = 0;
266 
267  MyCompileCallback callback;
268  if (!box->Compile(&callback, logic_string)) {
269  cerr << boost::format(_mut("Error: Could not compile logic file `%s'.")) % logicfile
270  << std::endl;
271  return 1;
272  }
273 
274  free(logic_string);
275  logic_string = NULL;
276 
277  route->SetActive(true);
278 
279  box -> Open();
280 
281  box -> KeyboardAnalysis(startkeys);
282  mutabor::InputDeviceClass::BatchPlay();
283 
284  in -> Close();
285  box -> Close();
286  out -> Close();
287 
288  if (out)
289  out -> Destroy();
290  out = NULL;
291  outguard.reset();
292  route -> Destroy();
293  route.reset();
294  box -> Destroy();
295  box.reset();
296  in -> Destroy();
297  in = NULL;
298  inguard.reset();
299  }
300  catch(const exception& e) {
301  cerr << "error: " << e.what() << "\n";
302  return 1;
303  }
304  catch(...) {
305  cerr << "Exception of unknown type!\n";
306  }
307 
308 
309  return 0;
