90 void boxTest::testAddKey()
91 {
92  // fill using a pattern of coprime numbers. So that we can check correct assumption about filling.
93  // group theory tells us that each modulus by division by MUT_BOX_MAX_KEY_INDEX+1 is contained an equal number of times.
94 
95  for (size_t i = 0 ; i < 7 * (MUT_BOX_MAX_KEY_INDEX+1) * 5; i++) {
96  int box = i % 7;
97  int noteidx = (i*5);
98  size_t last = boxes[box].last_key;
99  CPPUNIT_ASSERT( i >= 7 || last == 0);
100  CPPUNIT_ASSERT(boxes[box].key_count == (i/7));
101  mutabor_note_type * note = mutabor_create_key_in_box(&boxes[box]);
102  CPPUNIT_ASSERT(note!=NULL);
103  note->number = noteidx;
104  note->id = i/7;
105  CPPUNIT_ASSERT(boxes[box].key_count == (i/7) + 1);
106  CPPUNIT_ASSERT( i >= 7 || last == 0);
107  CPPUNIT_ASSERT(last != boxes[box].last_key || i < 7); // first row always returns 0 as well as empty box
108  CPPUNIT_ASSERT(boxes[box].last_key == (i/7));
109  // std::clog << i << std::endl;
110  verifyLastKey(i >= 6); // i < 6 as last iteration fulfils the condition
111  }
