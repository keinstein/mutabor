317  bool MutBoxShape::Detach( wxWindow *window )
318  {
319  // avoid deleting m_staticBox in our dtor if it's being detached from the
320  // sizer (which can happen because it's being already destroyed for
321  // example)
322  if ( window == m_icon )
323  {
324  UNREACHABLEC;
325  m_icon = NULL;
326  }
327 
328  bool ok = true;
329 
330  // reset the top-level parent's default item if it is this widget
331  wxTopLevelWindow *tlw =
332  wxDynamicCast(wxGetTopLevelParent((wxWindow*)this),
333  wxTopLevelWindow);
334 
335  if ( tlw )
336  {
337  wxWindow* tmpDefaultItem = tlw->GetTmpDefaultItem();
338  if ( tmpDefaultItem == this )
339  tlw->SetTmpDefaultItem(NULL);
340  else if ( tmpDefaultItem )
341  {
342  // A temporary default item masks the real
343  // default item, so
344  // temporarily unset the temporary default
345  // item so we can access the
346 
347  // real default item.
348  tlw->SetTmpDefaultItem(NULL);
349 
350  if ( tlw->GetDefaultItem() == this )
351  tlw->SetDefaultItem(NULL);
352 
353  // Set the temporary default item back.
354  tlw->SetTmpDefaultItem(tmpDefaultItem);
355  }
356  else if ( tlw->GetDefaultItem() == this )
357  tlw->SetDefaultItem(NULL);
358  }
359 
360  RemoveChild(window);
361  return ok && channels -> Detach( window );
362  }
