118 {
119  TRACEC;
120 
121  if ( !wxTextValidator::Validate(parent) )
122  return FALSE;
123 
124  wxTextCtrl *control = (wxTextCtrl *) m_validatorWindow ;
125 
126  wxString val(control->GetValue());
127 
128  bool ok = TRUE;
129 
130  wxString errormsg;
131 
132  long i;
133 
134  if ( Enabler && !Enabler->GetValue() ) {
135  // nix testen
136  } else if ( !val.ToLong(&i) ) {
137  if ( !Enabler || Enabler->GetValue() ) {
138  ok = FALSE;
139  errormsg = _("'%s' is not a number.");
140  }
141  } else if ( (Style & NV_MIN) && i < Min ) {
142  ok = FALSE;
143 
144  errormsg = _("'%s' is too small.");
145  } else if ( (Style & NV_MAX) && i > Max ) {
146  ok = FALSE;
147 
148  errormsg = _("'%s' is too big.");
149  }
150 
151  if ( !ok ) {
152  wxASSERT_MSG( !errormsg.empty(), _T("you forgot to set errormsg") );
153 
154  m_validatorWindow->SetFocus();
155 
156  wxString buf;
157  buf.Printf(errormsg, val.c_str());
158 
159  wxMessageBox(buf, _("Validation conflict"),
160  wxOK | wxICON_EXCLAMATION, parent);
161  }
162 
163  return ok;
164 }
