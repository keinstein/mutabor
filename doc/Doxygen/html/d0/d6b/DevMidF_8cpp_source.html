154  Track::base Track::ReadMessage() {
155  uint8_t status, metatype = midi::META_SEQUENCER_SPECIFIC;
156  if ((status = at(position)) & 0x80) {
157  position++;
158  if (midi::reset_running_status (status))
159  running_status = 0;
160  else if (store_running_status(status))
161  running_status = status;
162  } else status = running_status;
163  if (! (status & 0x80)) {
164  boost::throw_exception(invalid_status (gettext_noop("Invalid status byte")));
165  }
166 
167  size_t offset = 1;
168  // databytes may be nagative
169  int data_bytes = midi::get_data_size(status) - 1;
170  if (data_bytes < 0) {
171  mutASSERT(status == midi::SYSEX_START ||
172  status == midi::SYSEX_END ||
173  status == midi::META);
174  if (status == midi::META) {
175  metatype = at(position++);
176  offset++;
177  }
178  data_bytes = ReadInt();
179  }
180 
181  data_bytes += offset;
182  base retval(data_bytes,status);
183  if (status == midi::META)
184  retval[1] = metatype;
185 
186  for (size_t i = offset ; i < (size_t)data_bytes ; i++ )
187  retval [i] = at(position++); // should throw on access violation.

502  bool InputMidiFile::Open()
503  {
504  mutASSERT(!isOpen);
505  DEBUGLOG (midifile, _T("start"));
506  Tracks.clear();
507  char Header[5] = {0,0,0,0,0}; // storage for header mark
508  size_t nTrack;
509  DWORD l;
510  timing.reset();
511 
512 
513  // read file
514  mutOpenIFstream(is, Name);
515 
516  if ( mutStreamBad(is) ) {
517  DEBUGLOG (midifile, _T("Opening Stream failed"));
518  runtime_error(false,
519  _("Can not open Midi input file '%s'."),
520  (const mutChar *) Name.c_str());
521  goto error_cleanup;
522  }
523 
524  // Header Chunk
525  // Flawfinder: ignore
526  mutReadStream(is,Header, 4);
527  if (strcmp(Header,"MThd")) {
528  runtime_error(false,
529  _("File '%s' is not a valid midi file."),
530  (const mutChar *)Name.c_str()) ;
531  goto error_cleanup;
532  }
533 
534  l = mutabor::ReadLength(is);
535 
536  if (l!=6) {
537  runtime_error(false,
538  _("Unknown header (chunk length %d) in file '%s'."),
539  l,
540  (const mutChar *)Name.c_str());
541  goto error_cleanup;
542  }
543 
544  BYTE a, b;
545 
546  // file type
547  a = mutGetC(is); //mutGetC(is,a);
548  FileType = ((int)a << 8) + mutGetC(is); //mutGetC(is,FileType);
549  if (FileType > 3) {
550  runtime_error(false,
551  _("Unknown file typ %d in file '%s'."),
552  FileType,
553  (const mutChar *) Name.c_str());
554  goto error_cleanup;
555  }
556 
557  // number of tracks
558  a = mutGetC(is); //mutGetC(is,a);
559  b = mutGetC(is); // mutGetC(is,b);
560  nTrack = (((int)a) << 8) + b;
561 
562  // speed info
563  a = mutGetC(is); //mutGetC(is,a);
564  b = mutGetC(is); //mutGetC(is,b);
565  try {
566  timing.set_MIDI_tick_signature(a,b);
567  } catch (std::range_error e) {
568  runtime_error(false,
569  _("Midi file '%s' has corrupted timing information."),
570  (const mutChar *)Name.c_str()) ;
571  }
572  DEBUGLOG(midifile,
573  _T("File type: %d; Tracks: %d; Speed: %d Ticks/Qarter"),
574  FileType,
575  (int)nTrack,
576  (int)timing.get_ticks());
577 
578  // rest of header
579  DWORD i;
580 
581  for (i = 6; i < l && !mutStreamEOF(is) && mutStreamGood(is); i++ )
582  a = mutGetC(is);// mutGetC(is,a);
583 
584  // Tracks lesen
585  Tracks.resize(nTrack,Track(timing));
586  channel_data.resize(16*nTrack);
587  if (Tracks.empty()) {
588  runtime_error(false,
589  _("Could not allocate memory for the track list of file '%s'."),
590  (const mutChar *)Name.c_str()) ;
591  goto error_cleanup;
592  }
593 
594  for (i = 0; i < nTrack; i++ ) {
595  mutReadStream(is,Header, 4);
596  if (strcmp(Header,"MTrk")) {
597  runtime_error(false,_("File '%s' has a broken track header."),
598  (const mutChar *)Name.c_str()) ;
599  goto error_cleanup;
600  }
601 
602  l = mutabor::ReadLength(is);
603 
604  Tracks[i].clear();
605  try {
606  Tracks[i].resize(l);
607  } catch (std::bad_alloc e) {
608  Tracks[i].clear();
609  }
610 
611  if ( Tracks[i].size() < l) {
612  Mode = DeviceCompileError;
613  runtime_error(false,
614  _("Could not allocate data for track %d of MIDI input file '%s'."),
615  i, (const mutChar *)Name.c_str());
616  goto error_cleanup;
617  }
618 
619 
620  if ( l > 0x10000000) {
621  // in order to avoid problems with 32bit integers we read large tracks in 3 chunks
622 
623  mutReadStream(is, (char*)(Tracks[i].data()), l/3);
624  mutReadStream(is, (char*)(Tracks[i].data())+l/3, l/3);
625  mutReadStream(is, (char*)(Tracks[i].data())+l/3+l/3,l-l/3- l/3);
626  } else
627  mutReadStream(is, (char*)(Tracks[i].data()), l);
628 
629  if ( mutStreamBad(is) ) {
630  runtime_error(false,
631  _("The MIDI input file “%s” is corrupted. This has been detected while reading track %d."),
632  (const mutChar *)Name.c_str(),i);
633  goto error_cleanup;
634  }
635 
636  channel_data[i].Reset();
637  }
638 
639  mutCloseStream(is);
640 
641  return base::Open();
642 
643  error_cleanup:
644  Mode = DeviceCompileError;
645  isOpen = false;
646  mutCloseStream(is);
647  Tracks.clear();
648  return false;
649  }

680  mutint64 InputMidiFile::PrepareNextEvent()

701  minDelta = NewMinDelta;
702  return NewMinDelta;
