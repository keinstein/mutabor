781  {

805  microseconds InputMidiFile::ReadMidiProceed(size_t nr, microseconds deltatime)
806  {
807  microseconds Delta = Tracks[nr].GetDelta();
808  mutASSERT(Delta >= microseconds(-1000000));
809  microseconds time = deltatime;
810 
811  while ( time >= Delta ) {
812 
813  // Delta Time
814  mutASSERT(time >= Delta);
815  time -= Delta;
816  Tracks[nr].PassDelta(Delta);
817  Track::base message = Tracks[nr].ReadMessage();
818 
819  if (message.size() >= 2 && message[0] == midi::META) {
820  switch (message[1]) {
821  case midi::META_END_OF_TRACK:
822  Tracks[nr].Stop();
823  Delta = NO_DELTA();
824  break;
825  case midi::META_SET_TEMPO: {
826  size_t j,e;
827  set_tempo_event *ev
828  = static_cast<set_tempo_event*>
829  (create_event(message,-1));
830  if (!ev) {
831  break;
832  }
833  event guard = ev;
834 
835  // file type 0 must not have more than
836  // 1 track file we treat it like type
837  // one even though files with multiple
838  // tracks are corrupted. type 2 has
839  // independent tracks
840  if (FileType > 1) {
841  j = nr; e = nr+1;
842  } else {
843  j = 0; e = Tracks.size();
844  }
845  for (; j < e; j++ )
846  Tracks[j].SetQuarterDuration(ev->get_tempo(),
847  true,
848  j>nr ? deltatime - time: (microseconds::zero()));
849 
850  DEBUGLOG(midifile,
851  ("Change tempo to %ldμs per quarter (next event after %ld)"),
852  ev->get_tempo(), minDelta);
853  }
854 
855  }
856  }
857 
858  Proceed(message, nr, nr << 4);
859 
860  if (Delta == NO_DELTA())
861  return Delta;
862 
863  Delta = Tracks[nr].ReadDelta();
864 
865  DEBUGLOG(midifile,
866  ("Next event on Track %d after %ld μs"),
867  (int)nr, Delta);
868  }
869 #ifdef DEBUG
870  microseconds checktime =
871 #endif
872  Tracks[nr].PassDelta(time);
873 
874 #ifdef DEBUG
875  mutASSERT(checktime == Delta - time);
876 #endif
877  return Delta - time;
878  }
