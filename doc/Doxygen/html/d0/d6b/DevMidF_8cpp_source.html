523  Tracks.clear();

525  size_t nTrack;
526  uint32_t l;
527  timing.reset();
528 
529  if (Name.empty()) {
530  runtime_error(error,
531  _mut("You instructed me to open a file without file name. I can't do that."));
532  Mode = DeviceCompileError;

536  }

538  // read file
539  boost::filesystem::detail::utf8_codecvt_facet utf8;
540  boost::filesystem::path p;
541  p.assign(Name, utf8);
542 
543  boost::filesystem::ifstream is(p,
544  std::ios::in | std::ios::binary);
545  if ( is.bad() ) {
546  DEBUGLOG (midifile, "Opening Stream failed" );
547  runtime_error(error,
548  boost::str(boost::format(_mut("Can not open Midi input file '%s'."))

556  if (strcmp(Header,"MThd")) {
557  runtime_error(error,
558  boost::str(boost::format(_mut("File '%s' is not a valid midi file."))

579  boost::str(boost::format(_mut("Unknown file type %d in file '%s'."))

609  a = is.get();// mutGetC(is,a);
610

613  channel_data.resize(16*nTrack);

615  runtime_error(error,
616  boost::str(boost::format(_mut("Could not allocate memory for the track list of file '%s'."))

622  is.read(Header, 4);

624  runtime_error(error,
625  boost::str(boost::format(_mut("File '%s' has a broken track header."))

628  }

631 
632  Tracks[i].clear();

635  } catch (std::bad_alloc e) {

637  }

641  runtime_error(error,

645  }

647

652  is.read((char*)(Tracks[i].data())+l/3, l/3);
653  is.read((char*)(Tracks[i].data())+l/3+l/3,l-l/3- l/3);
654  } else

662  }

733  mutint64 InputMidiFile::ReadMidiProceed(size_t nr, mutint64 deltatime)
734  {
735  mutint64 Delta = Tracks[nr].GetDelta();
736  mutASSERT(Delta >= -1000000);
737  mutint64 time = deltatime;
738 
739  while ( time >= Delta ) {
740 
741  // Delta Time
742  mutASSERT(time >= Delta);
743  time -= Delta;
744  Tracks[nr].PassDelta(Delta);
745  Track::base message = Tracks[nr].ReadMessage();
746 
747  if (message.size() >= 2 && message[0] == midi::META) {
748  switch (message[1]) {
749  case midi::META_END_OF_TRACK:
750  Tracks[nr].Stop();
751  Delta = MUTABOR_NO_DELTA;
752  break;
753  case midi::META_SET_TEMPO: {
754  size_t j,e;
755  set_tempo_event *ev
756  = static_cast<set_tempo_event*>
757  (create_event(&message,-1));
758  if (!ev) {
759  break;
760  }
761  event guard = ev;
762 
763  // file type 0 must not have more than
764  // 1 track file we treat it like type
765  // one even though files with multiple
766  // tracks are corrupted. type 2 has
767  // independent tracks
768  if (FileType > 1) {
769  j = nr; e = nr+1;
770  } else {
771  j = 0; e = Tracks.size();
772  }
773  for (; j < e; j++ )
774  Tracks[j].SetQuarterDuration(ev->get_tempo(),
775  true,
776  j>nr ? deltatime - time: (mutint64)0);
777 
778  DEBUGLOG(midifile,
779  ("Change tempo to %ldŒºs per quarter (next event after %ld)"),
780  ev->get_tempo(), minDelta);
781  }
782 
783  }
784  }
785 
786  Proceed(&message, nr, nr << 4);
787 
788  if (Delta == MUTABOR_NO_DELTA)
789  return Delta;
790 
791  Delta = Tracks[nr].ReadDelta();
792 
793  DEBUGLOG(midifile,
794  ("Next event on Track %d after %ld Œºs"),
795  (int)nr, Delta);
796  }
797 #ifdef DEBUG
798  mutint64 checktime =
799 #endif
800  Tracks[nr].PassDelta(time);
801
