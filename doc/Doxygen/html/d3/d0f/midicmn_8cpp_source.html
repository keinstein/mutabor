286  } else if (bank_coarse != -1 && output_data.get_bank_coarse() != bank_coarse) {
287  set_sound = true;

290  }
291  if (set_sound && sound == -1)
292  sound = output_data.get_program();

296  output_data.set_controller(midi::BANK_COARSE,bank_coarse);
297  output_data.set_controller(midi::BANK_FINE,bank_fine);

320  int inkey,
321  int velocity,
322  RouteClass * r,
323  size_t id,
324  const ChannelData & input_channel_data)
325  {
326  mutASSERT(this->isOpen);
327  if (!ChannelFilter(r).check()) {
328  UNREACHABLEC;
329  return;
330  }
331

333  box.get(), inkey, velocity, (int)id);
334 // int free = 16, freeSus = r->OTo, freeVelocitycity = 64, freeSusVelocitycity = 64, s;
335 // DWORD p;
336  long freq;
337 
338  if ( !box ) {
339  freq = ((long)inkey) << 24;
340  } else
341  freq = box->get_frequency(inkey);
342 
343  if ( !freq ) return;
344 
345  channel_queue_type::const_iterator pos =
346  channel_queue.reserve_channel_filtered(ChannelFilter(r));

354  ChannelData & output_data = Cd[channel];
355  pitch_bend_type note = pitch_and_bend(freq);
356  int8_t controller_value;
357 
359  for (const int * i = midi::get_holds();
360  *i >= 0; i++) {
361  if ( output_data.get_controller(*i) !=
362  (controller_value
363  = input_channel_data.get_controller(*i)) ) {
364  if (controller_value >= 0) {
365  controller(channel,*i, controller_value);
366  output_data.set_controller(*i, controller_value);
367  }
368  }
369  }
370 
371  if ( note.bend != output_data.get_bend() ) {
372  pitch_bend(channel,note.bend);
373  output_data.set_bend(note.bend);
374 
375  // Switch off any hold pedal if necessary
376  for (const int * i = midi::get_holds();
377  *i >= 0; i++) {
378  if ( output_data.get_controller(*i) > midi::CONTROLLER_OFF) {
379  controller(channel,
380  *i,
381  midi::CONTROLLER_OFF);
382  output_data.set_controller(*i,
383  midi::CONTROLLER_OFF);
384  }
385  }
386  }
387 
388 
389  CopyProgramChange(input_channel_data, output_data, channel);
390 
391  ton_auf_kanal[channel].active = true;
392  ton_auf_kanal[channel].inkey = inkey;
393  ton_auf_kanal[channel].velocity = velocity;
394  ton_auf_kanal[channel].outkey = note;
395  ton_auf_kanal[channel].channel = r->get_session_id();
396  ton_auf_kanal[channel].midi_channel = channel;
397  ton_auf_kanal[channel].unique_id = id;
398  ton_auf_kanal[channel].tuned_key = freq;
399 
400  note_on(channel, note.pitch, velocity);
401 
402  do_UpdateControllers(channel, input_channel_data);

674  uint8_t command = p[pos] & midi::TYPE_MASK;
675  int channel = p[pos] & midi::CHANNEL_MASK;
676  switch (command) {
677  case midi::NOTE_OFF:
678  case midi::NOTE_ON:
679  case midi::AFTER_TOUCH:
680  case midi::CONTROLLER:
681  case midi::PITCH_BEND:
682  if (n-pos < 3)
683  return;
684  if (p[pos+1] & 0x80) {
685  pos++;
686  } else if (p[pos+2] & 0x80) {
687  pos += 2;
688  } else {
689  Out(channel, command, p[pos+1], p[pos+2]);
690  last_command = p[pos];
691  pos += 3;
692  datalength = 2;
693  }
694  break;
695  case midi::PROGRAM_CHANGE:
696  case midi::CHANNEL_PRESSURE:
697  if (n-pos < 2)
698  return;
699  if (p[pos+1] & 0x80) {
700  pos++;
701  } else {
702  Out(channel, command, p[pos+1]);
703  last_command = p[pos];
704  pos += 2;
705  datalength = 1;
706  }
707  break;
708 
709  case midi::SYSTEM:
710  if (!(channel & 0x08)) datalength = 0;
711  channel = -1;
712  switch (p[pos]) {
713  case midi::SYSEX_START:
714  {
715  size_t counter = pos + 1;
716  while (counter < n &&
717  p[counter] != midi::SYSEX_END) {
718  if (p[counter] & 0x80) {
719  break;
720  }
721  counter++;
722  }
723  if (p[pos] == midi::SYSEX_END) counter++;
724  // Dont interfere with sysex messages as some
725  // devices require proper timing
726  Out.SendSysEx(channel,&p[pos+1],p + counter);
727  pos = counter;
728  }
729  break;
730  case midi::QUARTER_FRAME:
731  case midi::SONG_SELECT:
732  if (n-pos < 2)
733  return;
734  if (p[pos+1] & 0x80) {
735  pos++;
736  } else {
737  Out(channel, command, p[pos+1]);
738  pos += 2;
739  }
740  break;
741  case midi::SONG_POSITION:
742  if (n-pos < 3)
743  return;
744  if (p[pos+1] & 0x80) {
745  pos++;
746  } else if (p[pos+2] & 0x80) {
747  pos += 2;
748  } else {
749  Out(channel, command, p[pos+1], p[pos+2]);
750  pos += 3;
751  }
752  break;
753 
754  case midi::SYSEX_END:
755  case midi::CLOCK:
756  case midi::TUNE_REQUEST:
757  case midi::TICK:
758  case midi::START_PLAY:
759  case midi::CONTINUE_PLAY:
760  case midi::STOP_PLAY:
761  case midi::ACTIVE_SENSE:
762  case midi::RESET:
763  Out.RawMsg(channel, p[pos]);
764  default:
765  pos++;
766  }
767  default:
768  command = last_command & midi::TYPE_MASK;
769  channel = last_command & midi::CHANNEL_MASK;
770  switch (datalength) {
771  case 2:
772  if (n-pos < 2)
773  return;
774  if (p[pos+1] & 0x80) {
775  pos++;
776  } else {
777  Out(channel,command,p[pos],p[pos+1]);
778  pos += 2;
779  }
780  break;
781  case 1:
782  Out(channel,command,p[pos++]);
783  break;
784  default:
785  // highest bit = 0 -> no command
786  // scan for the next command
787  pos++;
788  }
789  }

791  }
