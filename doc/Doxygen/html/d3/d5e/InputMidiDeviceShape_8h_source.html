96  void SetRouteType(RouteType routetype) {
97  wxPanel * panel;
98  switch (routetype) {
99  case mutabor::RTall: panel = all_panel; break;
100  case mutabor::RTchannel: panel = channel_panel; break;
101  case mutabor::RTstaff: panel = key_panel; break;
102  case mutabor::RTelse: panel = else_panel; break;
103  default:
104  UNREACHABLEC;
105  std::cerr << "MidiInputFilterPanel::SetRouteType: invaid route type "
106  << routetype << std::endl;
107  abort();
108  panel = NULL;
109  }
110  for (size_t i = 0 ; i < type->GetPageCount(); i++) {
111  if (type -> GetPage(i) != panel) continue;
112  type->SetSelection(i);
113  }
114  }
