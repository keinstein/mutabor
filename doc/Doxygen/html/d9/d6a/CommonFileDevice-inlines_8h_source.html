270  inline void CommonFileInputDevice::Play()
271  {
272  static bool starting = false;
273  ScopedLock modelock(lockMode);
274 
275  if (starting) return;
276  starting = true;
277 
278  switch (Mode) {
279  case DeviceCompileError:
280  case DeviceTimingError:
281  case DeviceUnregistered:
282  case DeviceKilled:
283  DEBUGLOG(timer,
284  ("Returnung due to device error."));
285  starting = false;
286  return;
287  case DeviceInitializing:
288  // this will eventually change to DeviceStop
289  // DevicePause case will wait until the device is initialized.
290  case DeviceStop:
291  referenceTime = 0;
292  pauseTime = 0;
293  case DevicePause:
294  DEBUGLOG(timer,
295  ("Paused. Realtime = %d."),
296  (int)CurrentTime.isRealtime());
297  // threadsignal |= ResetTime;
298  referenceTime += CurrentTime.Get() - pauseTime;
299  pauseTime = 0;
300 
301  // timer should be 0 in Realtime mode
302  mutASSERT(!timer || CurrentTime.isRealtime());
303  if (timer) {
304  threadsignal &= ~(int)RequestPause;
305  DEBUGLOG(thread,
306  ("Thread %p locking at threadsignal = %x"),
307  Thread::This(),
308  threadsignal.get());
309  waitMutex.Lock();
310  DEBUGLOG(thread,
311  ("Thread %p broadcasting at threadsignal = %x"),
312  Thread::This(),
313  threadsignal.get());
314  waitCondition.Broadcast();
315  waitMutex.Unlock();
316  }
317  break;
318  case DevicePlay:
319  starting = false;
320  return;
321 
322  }
323  DEBUGLOG (timer, "timer = %p" ,(void*)timer);
324  Mode = DevicePlay;
325  starting = false;
326  }

328  inline void CommonFileInputDevice::Pause()
329  {
330  switch (Mode) {
331  case DeviceCompileError:
332  case DeviceTimingError:
333  case DeviceStop:
334  break;
335  case DevicePlay:
336  if ( CurrentTime.isRealtime() ) {
337  // we use this function for some cleanup inside Stop()
338  // which is called from inside the thread.
339  // In this case we shouldn't pause ourselves
340  threadsignal |= RequestPause;
341  }
342  pauseTime = CurrentTime.Get();
343  Mode = DevicePause;
344  break;
345  case DevicePause:
346  Play(); // A mechanical Pause button usually is released if pressed twice
347  break;
348  default:
349  mutASSERT(!"Unknown value");
350  }
351  }

362  mutASSERT(timer != NULL);
363  mutASSERT(Mode != DeviceCompileError );
364 
365 
366 
367  if (Mode == DeviceInitializing)
368  Mode = DeviceStop;
369  /* tell the main thread that we are initialized */
370  DEBUGLOG(thread,
371  ("Thread %p locking at threadsignal = %x"),
372  Thread::This(),
373  threadsignal.get());
374  waitMutex.Lock();
375  DEBUGLOG(thread,
376  ("Thread %p broadcasting at threadsignal = %x"),
377  Thread::This(),
378  threadsignal.get());
379  waitCondition.Broadcast();
380  DEBUGLOG(thread,
381  ("Thread %p locking again at threadsignal = %x"),
382  Thread::This(),
383  threadsignal.get());
384  /* leave Mutex locked */
385  /*
386  waitMutex.Lock();
387  DEBUGLOG(thread,
388  ("Thread %p locked at threadsignal = %x"),
389  Thread::This(),
390  threadsignal.get());
391  */
392 
393  mutint64 nextEvent = (0); // in Œºs
394  mutint64 playTime = (0); // in Œºs
395  mutint64 reference = (0); // in Œºs
396  mutint64 delta = GetNO_DELTA(); // in ms
397  wxThread::ExitCode e = 0;
398  while (true) {
399  if (threadsignal & RequestPause) {
400  if (threadsignal & RequestPanic ) {
401  this->Panic(midi::DEFAULT_PANIC);
402  threadsignal &= ~(int)RequestPanic;
403  } else {
404  SilenceKeys(false);
405  }
406  DEBUGLOG(thread,
407  ("Thread %p waiting at threadsignal = %x"),
408  Thread::This(),
409  threadsignal.get());
410  waitCondition.Wait();
411  DEBUGLOG(thread,
412  ("Thread %p continuing and locking at threadsignal = %x"),
413  Thread::This(),
414  threadsignal.get());
415 #if 0
416  waitMutex.Lock();
417  DEBUGLOG(thread,
418  ("Thread %p Locked at threadsignal = %x"),
419  Thread::This(),
420  threadsignal.get());
421 #endif
422  if (threadsignal & RequestExit)
423  break;
424  if (threadsignal & ResetTime) {
425  nextEvent = (0); // in Œºs
426  playTime = (0); // in Œºs
427  doResetTime();
428  threadsignal &= ~(int)ResetTime;
429  }
430  reference = referenceTime;
431  ResumeKeys();
432  } else if (thistimer -> TestDestroy()) {
433  return (void *) (Mode + (size_t)0x100);
434  }
435  mutASSERT(IsDelta(nextEvent));
436  if (!delta && IsDelta(nextEvent)) {
437  nextEvent = PrepareNextEvent();
438  mutASSERT(nextEvent >= 0);
439  DEBUGLOG(timer,
440  ("Preparing next event is at %ld (isdelta=%d)"),
441  nextEvent,
442  IsDelta(nextEvent));
443  DEBUGLOG (timer, "NoDelta = %ld" , GetNO_DELTA());
444  if (IsDelta(nextEvent)) {
445  reference = referenceTime;
446  playTime += nextEvent;
447  }
448  }
449  if (!IsDelta(nextEvent)) {
450  {
451  ScopedLock modelock(lockMode);
452  switch (Mode) {
453  case DevicePlay:
454  case DevicePause:
455  case DeviceStop:
456  break;
457  case DeviceTimingError:
458  case DeviceCompileError:
459  default:
460  e = (void *)Mode;
461  break;
462  }
463  // unlocking
464  }
465  Stop();
466  } else if (!(threadsignal & (ResetTime | RequestExit | RequestPause))) {
467  DEBUGLOG (timer, "time: %ld Œºs" ,CurrentTime.Get());
468  delta = reference + playTime - CurrentTime.Get();
469  DEBUGLOG (timer, "Delta %ld Œºs." ,delta);
470 
471  /* we wake up in regular time intervals in order
472  to assure that external commands (stop/close)
473  can be handled in reasonable time)
474 
475  The delta slices should be smaller than the
476  resolution that is forced in the test suite.
477  Currently (as this comment is created) the
478  maximum allowed difference is 10 ms.
479  */
480  if (delta > 2000) {
481  CurrentTime.Sleep(2000);
482  } else {
483  if (delta > 0)
484  CurrentTime.Sleep(delta);
485  delta = 0;
486  }
487  }
488  }
489  DEBUGLOG (timer, "returning" );
490  switch (Mode) {
491  case DevicePlay:
492  case DevicePause:
493  Stop();
494  case DeviceStop:
495  break;
496  case DeviceTimingError:
497  case DeviceCompileError:
498  default:
499  e = (void *)Mode;
500  break;
501  }
502  DEBUGLOG(thread,
503  ("Thread %p unlocking at threadsignal = %x"),
504  Thread::This(),
505  threadsignal.get());
506  // assure that Close() has done its worke before exiting
507  exitLock.Lock();
508  threadCleanUp();
509  waitMutex.Unlock();
510  exitLock.Unlock();
511  return e;
512  }
