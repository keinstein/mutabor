332  inline void CommonFileInputDevice::Play()
333  {
334  static bool starting = false; // should be atomic
335 
336  if (starting) return;
337  starting = true;
338 
339  {
340  ScopedLock<CommonFileInputDevice> modelock(*this);
341  switch (Mode) {
342  case DeviceCompileError:
343  case DeviceTimingError:
344  case DeviceUnregistered:
345  case DeviceKilled:
346  DEBUGLOG(timer,
347  ("Returnung due to device error."));
348  starting = false;
349  return;
350  case DeviceInitializing:
351  // this will eventually change to DeviceStop
352  // DevicePause case will wait until the device is initialized.
353  case DeviceStop:
354  referenceTime = CurrentTimer::time_point();
355  pauseTime = CurrentTimer::time_point();
356  case DevicePause:
357  break;
358  case DevicePlay:
359  starting = false;
360  return;
361 
362  }
363 
364  DEBUGLOG(timer,
365  ("Paused. Realtime = %d."),
366  (int)CurrentTime.isRealtime());
367  // threadsignal |= ResetTime;
368  referenceTime = CurrentTimer::time_point(referenceTime) + ((CurrentTime.Get() - pauseTime));
369  pauseTime = CurrentTimer::time_point();
370  }
371 
372  // timer should be 0 in Realtime mode
373  // we must unlock *this so that the player can
374  // reach the synchronisation point
375  mutASSERT((!timer && !CurrentTime.isRealtime())
376  || (CurrentTime.isRealtime() && timer));
377  if (timer) {
378  thread_command &= ~(int)RequestPause; // threadsignal should be atomic
379  DEBUGLOG(thread,
380  ("Thread %s is changing thread command = %x"),
381  Thread::get_current_string_id().c_str(),
382  thread_command);
383  thread_command.notify_all();
384  ScopedLock<> lock(thread_state.get_mutex());
385  while (thread_state & RequestPause ||
386  ((thread_command & ResetTime) &&
387  !(thread_state & ResetTime))) {
388  thread_state.wait(lock);
389  }
390  thread_command &= ~(int)ResetTime;
391  thread_command.notify_all();
392  }
393  DEBUGLOG (timer, "timer = %p" ,(void*)timer);
394  {
395  ScopedLock<CommonFileInputDevice> modelock(*this);
396  Mode = DevicePlay;
397  }
398  /* we should not nest modeLock and playerActive, here. Otherwise we might deadlock. */
399 
400  starting = false;
401  //mutASSERT(timer);
402  }

404  inline void CommonFileInputDevice::Pause()
405  {
406  ScopedLock<CommonFileInputDevice> modelock(*this);
407  switch (Mode) {
408  case DeviceCompileError:
409  case DeviceTimingError:
410  case DeviceStop:
411  case DeviceKilled:
412  break;
413  case DevicePlay:
414  {
415  pauseTime = CurrentTime.Get();
416  if ( CurrentTime.isRealtime() ) {
417  // we use this function for some cleanup inside Stop()
418  // which is called from inside the thread.
419  // In this case we shouldn't pause ourselves
420  thread_command |= RequestPause;
421  }
422  thread_command.notify_all();
423  }
424  Mode = DevicePause;
425  break;
426  case DevicePause: {
427  unlock();
428  Play(); // A mechanical Pause button usually is released if pressed twice
429  lock();
430  }
431  break;
432  default:
433  mutASSERT(!"Unknown value");
434  }

440  inline int
441  CommonFileInputDevice::ThreadPlay(FileTimer * thistimer)
442  {
444  mutASSERT(Thread::get_current_id() == timer->get_id());
445 
446  mutASSERT(timer != NULL);
447  mutASSERT(Mode != DeviceCompileError );
448 
449 
450  {
451  ScopedLock<CommonFileInputDevice> modelock(*this);
452  if (Mode == DeviceInitializing)
453  Mode = DeviceStop;
454  }
455  DEBUGLOG(thread,
456  ("Thread %s locking at threadsignal = %x"),
457  Thread::get_current_string_id().c_str(),
458  thread_command);
459  /* We use manual locking and unlocking, here */
460  /* tell the main thread that we are initialized */
461  int e = 0;
462  {
463 
464  CurrentTimer::time_point nextEvent;
465  CurrentTimer::time_point reference;
466  microseconds playTime(0); // in s
467  microseconds eventDelta(0); // in s
468  try {
469  while (true) {
470  if (thread_command & RequestPause) {
471  if (thread_command & RequestPanic ) {
472  this->Panic(midi::DEFAULT_PANIC);
473  thread_state |= RequestPanic;
474  // TODO: Reset panic flag
475  } else {
476  SilenceKeys(false);
477  }
478  DEBUGLOG(thread,
479  ("Thread %s waiting at threadsignal = %x"),
480  Thread::get_current_string_id().c_str(),
481  thread_command);
482  try
483  {
484  thread_state &= ~(int) ResetTime;
485  thread_state |= RequestPause;
486  thread_state.notify_all();
487  ScopedLock<> lock(thread_command.get_mutex());
488  while ((thread_command & RequestPause) &&
489  !(thread_command & RequestExit))
490  thread_command.wait(lock);
491  if (!(thread_command & RequestPause)) {
492  thread_state &= ~(int)RequestPause;
493  thread_state.notify_all();
494  }
495  }
496  catch (const boost::thread_interrupted & e) {
497  Stop();
498  }
499  DEBUGLOG(thread,
500  ("Thread %s continuing and locking at threadsignal = %x"),
501  Thread::get_current_string_id().c_str(),
502  thread_command);
503  if (thread_command & ResetTime) {
504  eventDelta = microseconds::zero(); // in s
505  playTime = microseconds::zero(); // in s
506  doResetTime();
507  ScopedLock<> lock(thread_state.get_mutex());
508  thread_state |= ResetTime;
509  thread_state.notify_all();
510  }
511  reference = referenceTime;
512  nextEvent = reference + playTime;
513  ResumeKeys();
514  } else if (thistimer -> TestDestroy()) {
515  {
516  e = (Mode + (size_t)0x100);
517  // unlocking
518  }
519  break;
520  } else if (!IsDelta(eventDelta)) {
521  Stop();
522  } else if (!(thread_command & (ResetTime | RequestExit | RequestPause))) {
523  try {
524  CurrentTime.Sleep(nextEvent,thread_command);
525  }
526  catch (const boost::thread_interrupted & e) {
527  Stop();
528  }
529  }
530  // some shortcuts after the pause
531  if (thread_command & RequestExit) {
532  e = 0;
533  break;
534  }
535 
536  if (thread_command & RequestPause) {
537  continue;
538  }
539 
540  mutASSERT(IsDelta(eventDelta));
541  // we must evaluate the time again as spurious
542  // wakeups may occur and we wake up the thread
543  // on certain status changes. Furthermore the
544  // delay from the previous code also take some
545  // time.
546  if (nextEvent <= CurrentTime.Get()) {
547  try {
548  eventDelta = PrepareNextEvent();
549  } catch (...) {
550  exception_error();
551  }
552  mutASSERT(eventDelta >= microseconds::zero());
553  DEBUGLOG(timer,
554  ("Preparing next event is at %ld (isdelta=%d)"),
555  eventDelta,
556  IsDelta(eventDelta));
557  DEBUGLOG (timer, "NoDelta = %ld" , NO_DELTA());
558  if (IsDelta(eventDelta)) {
559  playTime += eventDelta;
560  nextEvent += eventDelta;
561  }
562  }
563  }
564  } catch (const boost::thread_interrupted & e) {}
565  }
566  DEBUGLOG (timer, "returning" );
567  {
568  ScopedLock<CommonFileInputDevice> modelock(*this);
569  switch (Mode) {
570  case DevicePlay:
571  case DevicePause:
572  case DeviceKilled:
573  Stop();
574  case DeviceStop:
575  break;
576  case DeviceTimingError:
577  case DeviceCompileError:
578  default:
579  break;
580  }
581  }
582  DEBUGLOG(thread,
583  ("Thread %s unlocking at threadsignal = %x"),
584  Thread::get_current_string_id().c_str(),
585  thread_command);
586  // assure that Close() has done its worke before exiting
587  threadCleanUp();
588  return e;
589  }
