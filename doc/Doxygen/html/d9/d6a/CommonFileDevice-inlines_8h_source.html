271  inline void CommonFileInputDevice::Play()
272  {
273  static bool starting = false;
274  ScopedLock modelock(lockMode);
275 
276  if (starting) return;
277  starting = true;
278 
279  switch (Mode) {
280  case DeviceCompileError:
281  case DeviceTimingError:
282  case DeviceUnregistered:
283  case DeviceKilled:
284  DEBUGLOG(timer,
285  _T("Returnung due to device error."));
286  starting = false;
287  return;
288  case DeviceInitializing:
289  // this will eventually change to DeviceStop
290  // DevicePause case will wait until the device is initialized.
291  case DeviceStop:
292  referenceTime = 0;
293  pauseTime = 0;
294  case DevicePause:
295  DEBUGLOG(timer,
296  _T("Paused. Realtime = %d."),
297  (int)CurrentTime.isRealtime());
298  // threadsignal |= ResetTime;
299  referenceTime += CurrentTime.Get() - pauseTime;
300  pauseTime = 0;
301 
302  // timer should be 0 in Realtime mode
303  mutASSERT(!timer || CurrentTime.isRealtime());
304  if (timer) {
305  threadsignal &= ~(int)RequestPause;
306  DEBUGLOG(thread,
307  _T("Thread %p locking at threadsignal = %x"),
308  Thread::This(),
309  threadsignal.get());
310  waitMutex.Lock();
311  DEBUGLOG(thread,
312  _T("Thread %p broadcasting at threadsignal = %x"),
313  Thread::This(),
314  threadsignal.get());
315  waitCondition.Broadcast();
316  waitMutex.Unlock();
317  }
318  break;
319  case DevicePlay:
320  starting = false;
321  return;
322 
323  }
324  DEBUGLOG(timer,_T("timer = %p"),(void*)timer);
325  Mode = DevicePlay;
326  starting = false;

329  inline void CommonFileInputDevice::Pause()
330  {
331  switch (Mode) {
332  case DeviceCompileError:
333  case DeviceTimingError:
334  case DeviceStop:
335  break;
336  case DevicePlay:
337  if ( CurrentTime.isRealtime() ) {
338  // we use this function for some cleanup inside Stop()
339  // which is called from inside the thread.
340  // In this case we shouldn't pause ourselves
341  threadsignal |= RequestPause;
342  }
343  pauseTime = CurrentTime.Get();
344  Mode = DevicePause;
345  break;
346  case DevicePause:
347  Play(); // A mechanical Pause button usually is released if pressed twice
348  break;
349  default:
350  mutASSERT(!"Unknown value");
351  }
352  }
