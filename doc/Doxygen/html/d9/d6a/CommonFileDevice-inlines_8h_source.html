271  {

273  ScopedLock modelock(lockMode);
274 
275  if (starting) return;

277 
278  switch (Mode) {
279  case DeviceCompileError:
280  case DeviceTimingError:

284  starting = false;

295  pauseTime = 0;

298  mutASSERT(!timer || CurrentTime.isRealtime());

300  threadsignal &= ~(int)RequestPause;

303  Thread::This(),

307  _T("Thread %p broadcasting at threadsignal = %x"),
308  Thread::This(),

311  waitMutex.Unlock();
312  }
313  break;
314  case DevicePlay:
315  starting = false;

324  inline void CommonFileInputDevice::Pause()
325  {
326  switch (Mode) {
327  case DeviceCompileError:
328  case DeviceTimingError:
329  case DeviceStop:
330  break;
331  case DevicePlay:
332  if ( CurrentTime.isRealtime() ) {
333  // we use this function for some cleanup inside Stop()
334  // which is called from inside the thread.
335  // In this case we shouldn't pause ourselves
336  threadsignal |= RequestPause;
337  }
338  pauseTime = CurrentTime.Get();
339  Mode = DevicePause;
340  break;
341  case DevicePause:
342  Play(); // A mechanical Pause button usually is released if pressed twice
343  break;
344  default:
345  mutASSERT(!"Unknown value");
346  }
347  }
