165  static void AllSetPlaystate(playstate pst = Stop, bool force = false) {
166  static playstate oldpst = Hide;
167  if (force) {
168  AllDoSetPlaystate(pst);
169  return;
170  }
171  if (pst == oldpst) return;
172  switch (pst) {
173  case Play:
174  case Hide:
175  oldpst = pst;
176  AllDoSetPlaystate(pst);
177  return;
178  break;
179  case Pause:
180  if (oldpst == Stop) {
181  oldpst = pst;
182  AllDoSetPlaystate(pst);
183  return;
184  }
185  break;
186  case Stop:
187  break;
188  default:
189  UNREACHABLECT(StatusBar);
190  return;
191  }
192  playstate newpst = GetCurrentPlayState();
193  if (newpst == oldpst) return;
194  oldpst = newpst;
195  AllDoSetPlaystate(newpst);
196  }
