166  static void AllSetPlaystate(playstate pst = Stop, bool force = false) {
167  static playstate oldpst = Hide;
168  if (force) {
169  AllDoSetPlaystate(pst);
170  return;
171  }
172  if (pst == oldpst) return;
173  switch (pst) {
174  case Play:
175  case Hide:
176  oldpst = pst;
177  AllDoSetPlaystate(pst);
178  return;
179  case Pause:
180  if (oldpst == Stop) {
181  oldpst = pst;
182  AllDoSetPlaystate(pst);
183  return;
184  }
185  break;
186  case Stop:
187  break;
188  default:
189  UNREACHABLECT(StatusBar);
190  return;
191  }
192  playstate newpst = GetCurrentPlayState();
193  if (newpst == oldpst) return;
194  oldpst = newpst;
195  AllDoSetPlaystate(newpst);
196  }
